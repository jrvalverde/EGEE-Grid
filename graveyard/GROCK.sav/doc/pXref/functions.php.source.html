  18  /**
  19   * open a connection to the Grid UI server
  20   * 
  21   *     The Grid User Interface server is the entry point to the Grid
  22   *  for users and user applications. This is where jobs are launched from.
  23   * 
  24   *     GridGRAMM has been designed to be able to be installed in any
  25   *  host, independent of whether it is an UI or not. Thus, to be able to
  26   *  submit jobs to the Grid, the server hosting GridGRAMM must connect to
  27   *  a Grid UI host to do the work.
  28   * 
  29   *     This routine opens a connection to a Grid UI host using a predefined
  30   *  username (i.e. all jobs will be run under said username). To allow the
  31   *  caller to communicate with the remote end, it creates two pipes, one for
  32   *  input and one for output, and redirect error messages to a file.
  33   * 
  34   *     These pipes lead to a child process that actually manages the 
  35   *  communication. Using a child process has several advantages: it simplifies
  36   *  the communication process by offloading the comm. logic to a separate,
  37   *  convenience tool, and by being able to use SSH as the child, we can increase
  38   *  security taking advantage of its encryption capabilities.
  39   * 
  40   *     The panorama therefore will look like this:
  41   * 
  42   *     HTML front-end --> processor.php <--> SSH <--> remote host <--> Grid
  43   * 
  44   *     @param server    A username@remove.grid.ui.host pair to use a command
  45   *             processor
  46   *     @param grid_path Path to working directory in the local machine
  47   *     @param process    A handle to the child process that manages communication
  48   *             with the Grid access point
  49   *     @param descriptorspec
  50   *     @param pipes    An array of file descriptors to be used to communicate
  51   *             withe the Grid UI through the child process.
  52   */
  53  function open_connection($server, $grid_path, $process, $descriptorspec, $pipes)
  54  {
  55      // The ssh connection is a 'child' process, in the User Interface 
  56      // opened wit 'proc_open'
  57      
  58      global $process, $descriptorspec, $pipes;
  59      $descriptorspec = array(
  60              0 => array("pipe", "r"),  // stdin is a pipe that the child will read from
  61              1 => array("pipe", "w"),  // stdout is a pipe that the child will write to
  62              2 => array("file", "$grid_path/error-output.txt", "a") // stderr is a file to write to
  63      );
  64      
  65      // Open the child process with 'proc_open' function. SSH connection with
  66      //  -x parameter, we need to disable the X11 forwarding.
  67      // $process = proc_open("ssh -t -t $server", $descriptorspec, $pipes);
  68      // Is required to us twice the parameter -t, otherwise we get the error:
  69      // "Pseudo-terminal will not be allocated because stdin is not a terminal".
  70      $process = proc_open("ssh -x -t -t $server", $descriptorspec, $pipes);
  71      
  72      // $pipes now looks like this:
  73      //   0 => writeable handle connected to child stdin
  74      //   1 => readable handle connected to child stdout
  75      // Any error output will be appended to $grid_path/error-output.txt
  76  
  77  
  78      if (!is_resource($process)) 
  79      {
  80          echo "connection error!!!<br \></body></html>\n";
  81          exit;
  82      }
  83  
  84  }

  87  /**
  88   * Close the connection with the remote Grid access point (UI node)
  89   *
  90   *     What we are going to do is to close the communication pipes
  91   *  and kill the child processes that actually handles communication
  92   *  with the remote Grid UI host (ssh).
  93   * 
  94   *  @note CLOSE_CONNECTION IS USEFUL TO DEBUG!!!
  95   *
  96   *  @param $process
  97   *  @param $descriptorspec
  98   *  @param $pipes
  99   */
 100  function close_connection($process, $descriptorspec, $pipes)
 101  {
 102      
 103      // Required to close the 'child' process, otherwise 'proc_close' function hangs!!!
 104      fwrite($pipes[0], "exit\n");
 105  
 106      fclose($pipes[0]);
 107  
 108      // The output from the commands
 109      // Uncomment to debug the 'child' process!!!
 110      /*
 111      while (!feof($pipes[1]))
 112      {
 113          echo fgets($pipes[1], 1024)."<br />";
 114      }
 115      */
 116      fclose($pipes[1]);
 117  
 118      // It's important that you close any pipes before calling
 119      // proc_close in order to avoid a deadlock
 120      $return_value = proc_close($process);
 121      
 122      // echo "\ncommand returned $return_value\n";    
 123  }

 184  function enter_grid($pipes, $grid_pass)

 197  function create_working_directory($grid_path)
 198  {
 199      // Random value for the directory
 200      global $random_value;
 201      srand((double)microtime()*10000);
 202      $random_value = rand();
 203      
 204      // Working directory in the local machine
 205      if (!mkdir("$grid_path/$random_value", 0777))
 206      {
 207          echo "Error, can't generate the working directory<br /></body></html>\n";
 208          exit;    
 209      }
 210      // Mkdir don't work properly with the permissions
 211      chmod( "$grid_path/$random_value", 0777 );
 212      
 213      // Gramm directory in the local machine
 214      if (!mkdir ("$grid_path/$random_value/gramm", 0777))
 215      {
 216          echo "Error, can't generate the gramm directory<br /></body></html>\n";
 217          exit;    
 218      }
 219      // Mkdir don't work properly with the permissions
 220      chmod( "$grid_path/$random_value/gramm", 0777 );
