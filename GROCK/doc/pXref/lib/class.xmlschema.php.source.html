  86      function parseFile($xml,$type){
  87          // parse xml file

  88          if($xml != ""){
  89              $xmlStr = @join("",@file($xml));
  90              if($xmlStr == ""){
  91                  $msg = 'Error reading XML from '.$xml;
  92                  $this->setError($msg);
  93                  $this->debug($msg);
  94              return false;
  95              } else {
  96                  $this->debug("parsing $xml");
  97                  $this->parseString($xmlStr,$type);
  98                  $this->debug("done parsing $xml");
  99              return true;
 100              }
 101          }
 102          return false;
 103      }

 105      /**

 106      * parse an XML string

 107      *

 108      * @param    string $xml path or URL

 109      * @param string $type, (schema|xml)

 110      * @access   private

 111      */
 112      function parseString($xml,$type){
 113          // parse xml string

 114          if($xml != ""){
 115  
 116              // Create an XML parser.

 117              $this->parser = xml_parser_create();
 118              // Set the options for parsing the XML data.

 119              xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING, 0);
 120  
 121              // Set the object for the parser.

 122              xml_set_object($this->parser, $this);
 123  
 124              // Set the element handlers for the parser.

 125              if($type == "schema"){
 126                  xml_set_element_handler($this->parser, 'schemaStartElement','schemaEndElement');
 127                  xml_set_character_data_handler($this->parser,'schemaCharacterData');
 128              } elseif($type == "xml"){
 129                  xml_set_element_handler($this->parser, 'xmlStartElement','xmlEndElement');
 130                  xml_set_character_data_handler($this->parser,'xmlCharacterData');
 131              }
 132  
 133              // Parse the XML file.

 134              if(!xml_parse($this->parser,$xml,true)){
 135              // Display an error message.

 136                  $errstr = sprintf('XML error parsing XML schema on line %d: %s',
 137                  xml_get_current_line_number($this->parser),
 138                  xml_error_string(xml_get_error_code($this->parser))
 139                  );
 140                  $this->debug($errstr);
 141                  $this->debug("XML payload:\n" . $xml);
 142                  $this->setError($errstr);
 143              }
 144              
 145              xml_parser_free($this->parser);
 146          } else{
 147              $this->debug('no xml passed to parseString()!!');
 148              $this->setError('no xml passed to parseString()!!');
 149          }
 150      }

 152      /**

 153      * start-element handler

 154      *

 155      * @param    string $parser XML parser object

 156      * @param    string $name element name

 157      * @param    string $attrs associative array of attributes

 158      * @access   private

 159      */
 160      function schemaStartElement($parser, $name, $attrs) {
 161          
 162          // position in the total number of elements, starting from 0

 163          $pos = $this->position++;
 164          $depth = $this->depth++;
 165          // set self as current value for this depth

 166          $this->depth_array[$depth] = $pos;
 167          $this->message[$pos] = array('cdata' => ''); 
 168          if ($depth > 0) {
 169              $this->defaultNamespace[$pos] = $this->defaultNamespace[$this->depth_array[$depth - 1]];
 170          } else {
 171              $this->defaultNamespace[$pos] = false;
 172          }
 173  
 174          // get element prefix

 175          if($prefix = $this->getPrefix($name)){
 176              // get unqualified name

 177              $name = $this->getLocalPart($name);
 178          } else {
 179              $prefix = '';
 180          }
 181          
 182          // loop thru attributes, expanding, and registering namespace declarations

 183          if(count($attrs) > 0){
 184              foreach($attrs as $k => $v){
 185                  // if ns declarations, add to class level array of valid namespaces

 186                  if(ereg("^xmlns",$k)){
 187                      //$this->xdebug("$k: $v");

 188                      //$this->xdebug('ns_prefix: '.$this->getPrefix($k));

 189                      if($ns_prefix = substr(strrchr($k,':'),1)){
 190                          //$this->xdebug("Add namespace[$ns_prefix] = $v");

 191                          $this->namespaces[$ns_prefix] = $v;
 192                      } else {
 193                          $this->defaultNamespace[$pos] = $v;
 194                          if (! $this->getPrefixFromNamespace($v)) {
 195                              $this->namespaces['ns'.(count($this->namespaces)+1)] = $v;
 196                          }
 197                      }
 198                      if($v == 'http://www.w3.org/2001/XMLSchema' || $v == 'http://www.w3.org/1999/XMLSchema'){
 199                          $this->XMLSchemaVersion = $v;
 200                          $this->namespaces['xsi'] = $v.'-instance';
 201                      }
 202                  }
 203              }
 204              foreach($attrs as $k => $v){
 205                  // expand each attribute

 206                  $k = strpos($k,':') ? $this->expandQname($k) : $k;
 207                  $v = strpos($v,':') ? $this->expandQname($v) : $v;
 208                  $eAttrs[$k] = $v;
 209              }
 210              $attrs = $eAttrs;
 211          } else {
 212              $attrs = array();
 213          }
 214          // find status, register data

 215          switch($name){
 216              case 'all':
 217              case 'choice':
 218              case 'sequence':
 219                  //$this->xdebug("compositor $name for currentComplexType: $this->currentComplexType and currentElement: $this->currentElement");

 220                  $this->complexTypes[$this->currentComplexType]['compositor'] = $name;
 221                  if($name == 'all' || $name == 'sequence'){
 222                      $this->complexTypes[$this->currentComplexType]['phpType'] = 'struct';
 223                  }
 224              break;
 225              case 'attribute':
 226                  //$this->xdebug("parsing attribute $attrs[name] $attrs[ref] of value: ".$attrs['http://schemas.xmlsoap.org/wsdl/:arrayType']);

 227                  $this->xdebug("parsing attribute " . $this->varDump($attrs));
 228                  if (isset($attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'])) {
 229                      $v = $attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'];
 230                      if (!strpos($v, ':')) {
 231                          // no namespace in arrayType attribute value...

 232                          if ($this->defaultNamespace[$pos]) {
 233                              // ...so use the default

 234                              $attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'] = $this->defaultNamespace[$pos] . ':' . $attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'];
 235                          }
 236                      }
 237                  }
 238                  if(isset($attrs['name'])){
 239                      $this->attributes[$attrs['name']] = $attrs;
 240                      $aname = $attrs['name'];
 241                  } elseif(isset($attrs['ref']) && $attrs['ref'] == 'http://schemas.xmlsoap.org/soap/encoding/:arrayType'){
 242                      if (isset($attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'])) {
 243                          $aname = $attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'];
 244                      } else {
 245                          $aname = '';
 246                      }
 247                  } elseif(isset($attrs['ref'])){
 248                      $aname = $attrs['ref'];
 249                      $this->attributes[$attrs['ref']] = $attrs;
 250                  }
 251                  
 252                  if(isset($this->currentComplexType)){
 253                      $this->complexTypes[$this->currentComplexType]['attrs'][$aname] = $attrs;
 254                  } elseif(isset($this->currentElement)){
 255                      $this->elements[$this->currentElement]['attrs'][$aname] = $attrs;
 256                  }
 257                  // arrayType attribute

 258                  if(isset($attrs['http://schemas.xmlsoap.org/wsdl/:arrayType']) || $this->getLocalPart($aname) == 'arrayType'){
 259                      $this->complexTypes[$this->currentComplexType]['phpType'] = 'array';
 260                      $prefix = $this->getPrefix($aname);
 261                      if(isset($attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'])){
 262                          $v = $attrs['http://schemas.xmlsoap.org/wsdl/:arrayType'];
 263                      } else {
 264                          $v = '';
 265                      }
 266                      if(strpos($v,'[,]')){
 267                          $this->complexTypes[$this->currentComplexType]['multidimensional'] = true;
 268                      }
 269                      $v = substr($v,0,strpos($v,'[')); // clip the []
 270                      if(!strpos($v,':') && isset($this->typemap[$this->XMLSchemaVersion][$v])){
 271                          $v = $this->XMLSchemaVersion.':'.$v;
 272                      }
 273                      $this->complexTypes[$this->currentComplexType]['arrayType'] = $v;
 274                  }
 275              break;
 276              case 'complexType':
 277                  if(isset($attrs['name'])){
 278                      $this->xdebug('processing named complexType '.$attrs['name']);
 279                      $this->currentElement = false;
 280                      $this->currentComplexType = $attrs['name'];
 281                      $this->complexTypes[$this->currentComplexType] = $attrs;
 282                      $this->complexTypes[$this->currentComplexType]['typeClass'] = 'complexType';
 283                      if(isset($attrs['base']) && ereg(':Array$',$attrs['base'])){
 284                          $this->complexTypes[$this->currentComplexType]['phpType'] = 'array';
 285                      } else {
 286                          $this->complexTypes[$this->currentComplexType]['phpType'] = 'struct';
 287                      }
 288                  }else{
 289                      $this->xdebug('processing unnamed complexType for element '.$this->currentElement);
 290                      $this->currentComplexType = $this->currentElement . '_ContainedType';
 291                      $this->currentElement = false;
 292                      $this->complexTypes[$this->currentComplexType] = $attrs;
 293                      $this->complexTypes[$this->currentComplexType]['typeClass'] = 'complexType';
 294                      if(isset($attrs['base']) && ereg(':Array$',$attrs['base'])){
 295                          $this->complexTypes[$this->currentComplexType]['phpType'] = 'array';
 296                      } else {
 297                          $this->complexTypes[$this->currentComplexType]['phpType'] = 'struct';
 298                      }
 299                  }
 300              break;
 301              case 'element':
 302                  // elements defined as part of a complex type should

 303                  // not really be added to $this->elements, but for some

 304                  // reason, they are

 305                  if(isset($attrs['type'])){
 306                      $this->xdebug("processing typed element ".$attrs['name']." of type ".$attrs['type']);
 307                      $this->currentElement = $attrs['name'];
 308                      $this->elements[ $attrs['name'] ] = $attrs;
 309                      $this->elements[ $attrs['name'] ]['typeClass'] = 'element';
 310                      if (!isset($this->elements[ $attrs['name'] ]['form'])) {
 311                          $this->elements[ $attrs['name'] ]['form'] = $this->schemaInfo['elementFormDefault'];
 312                      }
 313                      $ename = $attrs['name'];
 314                  } elseif(isset($attrs['ref'])){
 315                      $ename = $attrs['ref'];
 316                  } else {
 317                      $this->xdebug("processing untyped element ".$attrs['name']);
 318                      $this->currentElement = $attrs['name'];
 319                      $this->elements[ $attrs['name'] ] = $attrs;
 320                      $this->elements[ $attrs['name'] ]['typeClass'] = 'element';
 321                      $this->elements[ $attrs['name'] ]['type'] = $this->schemaTargetNamespace . ':' . $attrs['name'] . '_ContainedType';
 322                      if (!isset($this->elements[ $attrs['name'] ]['form'])) {
 323                          $this->elements[ $attrs['name'] ]['form'] = $this->schemaInfo['elementFormDefault'];
 324                      }
 325                  }
 326                  if(isset($ename) && $this->currentComplexType){
 327                      $this->complexTypes[$this->currentComplexType]['elements'][$ename] = $attrs;
 328                  }
 329              break;
 330              // we ignore enumeration values

 331              //case 'enumeration':

 332              //break;

 333              case 'import':
 334                  if (isset($attrs['schemaLocation'])) {
 335                      //$this->xdebug('import namespace ' . $attrs['namespace'] . ' from ' . $attrs['schemaLocation']);

 336                      $this->imports[$attrs['namespace']][] = array('location' => $attrs['schemaLocation'], 'loaded' => false);
 337                  } else {
 338                      //$this->xdebug('import namespace ' . $attrs['namespace']);

 339                      $this->imports[$attrs['namespace']][] = array('location' => '', 'loaded' => true);
 340                      if (! $this->getPrefixFromNamespace($attrs['namespace'])) {
 341                          $this->namespaces['ns'.(count($this->namespaces)+1)] = $attrs['namespace'];
 342                      }
 343                  }
 344              break;
 345              case 'restriction':
 346                  //$this->xdebug("in restriction for currentComplexType: $this->currentComplexType and currentElement: $this->currentElement");

 347                  if($this->currentElement){
 348                      $this->elements[$this->currentElement]['type'] = $attrs['base'];
 349                  } elseif($this->currentSimpleType){
 350                      $this->simpleTypes[$this->currentSimpleType]['type'] = $attrs['base'];
 351                  } elseif($this->currentComplexType){
 352                      $this->complexTypes[$this->currentComplexType]['restrictionBase'] = $attrs['base'];
 353                      if(strstr($attrs['base'],':') == ':Array'){
 354                          $this->complexTypes[$this->currentComplexType]['phpType'] = 'array';
 355                      }
 356                  }
 357              break;
 358              case 'schema':
 359                  $this->schemaInfo = $attrs;
 360                  $this->schemaInfo['schemaVersion'] = $this->getNamespaceFromPrefix($prefix);
 361                  if (isset($attrs['targetNamespace'])) {
 362                      $this->schemaTargetNamespace = $attrs['targetNamespace'];
 363                  }
 364                  if (!isset($attrs['elementFormDefault'])) {
 365                      $this->schemaInfo['elementFormDefault'] = 'unqualified';
 366                  }
 367              break;
 368              case 'simpleType':
 369                  if(isset($attrs['name'])){
 370                      $this->xdebug("processing simpleType for name " . $attrs['name']);
 371                      $this->currentSimpleType = $attrs['name'];
 372                      $this->simpleTypes[ $attrs['name'] ] = $attrs;
 373                      $this->simpleTypes[ $attrs['name'] ]['typeClass'] = 'simpleType';
 374                      $this->simpleTypes[ $attrs['name'] ]['phpType'] = 'scalar';
 375                  } else {
 376                      //echo 'not parsing: '.$name;

 377                      //var_dump($attrs);

 378                  }
 379              break;
 380              default:
 381                  //$this->xdebug("do not have anything to do for element $name");

 382          }
 383      }

 386      * end-element handler

 387      *

 388      * @param    string $parser XML parser object

 389      * @param    string $name element name

 390      * @access   private

 391      */
 392      function schemaEndElement($parser, $name) {
 393          // bring depth down a notch

 394          $this->depth--;
 395          // position of current element is equal to the last value left in depth_array for my depth

 396          if(isset($this->depth_array[$this->depth])){
 397              $pos = $this->depth_array[$this->depth];
 398          }
 399          // move on...

 400          if($name == 'complexType'){
 401              $this->currentComplexType = false;
 402              $this->currentElement = false;
 403          }
 404          if($name == 'element'){
 405              $this->currentElement = false;
 406          }
 407          if($name == 'simpleType'){
 408              $this->currentSimpleType = false;
 409          }
 410      }

 424      /**

 425      * serialize the schema

 426      *

 427      * @access   public

 428      */
 429      function serializeSchema(){
 430  
 431          $schemaPrefix = $this->getPrefixFromNamespace($this->XMLSchemaVersion);
 432          $xml = '';
 433          // imports

 434          if (sizeof($this->imports) > 0) {
 435              foreach($this->imports as $ns => $list) {
 436                  foreach ($list as $ii) {
 437                      if ($ii['location'] != '') {
 438                          $xml .= " <$schemaPrefix:import location=\"" . $ii['location'] . '" namespace="' . $ns . "\" />\n";
 439                      } else {
 440                          $xml .= " <$schemaPrefix:import namespace=\"" . $ns . "\" />\n";
 441                      }
 442                  }
 443              } 
 444          } 
 445          // complex types

 446          foreach($this->complexTypes as $typeName => $attrs){
 447              $contentStr = '';
 448              // serialize child elements

 449              if(isset($attrs['elements']) && (count($attrs['elements']) > 0)){
 450                  foreach($attrs['elements'] as $element => $eParts){
 451                      if(isset($eParts['ref'])){
 452                          $contentStr .= "   <$schemaPrefix:element ref=\"$element\"/>\n";
 453                      } else {
 454                          $contentStr .= "   <$schemaPrefix:element name=\"$element\" type=\"" . $this->contractQName($eParts['type']) . "\"/>\n";
 455                      }
 456                  }
 457              }
 458              // attributes

 459              if(isset($attrs['attrs']) && (count($attrs['attrs']) >= 1)){
 460                  foreach($attrs['attrs'] as $attr => $aParts){
 461                      $contentStr .= "    <$schemaPrefix:attribute ref=\"".$this->contractQName($aParts['ref']).'"';
 462                      if(isset($aParts['http://schemas.xmlsoap.org/wsdl/:arrayType'])){
 463                          $this->usedNamespaces['wsdl'] = $this->namespaces['wsdl'];
 464                          $contentStr .= ' wsdl:arrayType="'.$this->contractQName($aParts['http://schemas.xmlsoap.org/wsdl/:arrayType']).'"';
 465                      }
 466                      $contentStr .= "/>\n";
 467                  }
 468              }
 469              // if restriction

 470              if( isset($attrs['restrictionBase']) && $attrs['restrictionBase'] != ''){
 471                  $contentStr = "   <$schemaPrefix:restriction base=\"".$this->contractQName($attrs['restrictionBase'])."\">\n".$contentStr."   </$schemaPrefix:restriction>\n";
 472              }
 473              // compositor obviates complex/simple content

 474              if(isset($attrs['compositor']) && ($attrs['compositor'] != '')){
 475                  $contentStr = "  <$schemaPrefix:$attrs[compositor]>\n".$contentStr."  </$schemaPrefix:$attrs[compositor]>\n";
 476              }
 477              // complex or simple content

 478              elseif((isset($attrs['elements']) && count($attrs['elements']) > 0) || (isset($attrs['attrs']) && count($attrs['attrs']) > 0)){
 479                  $contentStr = "  <$schemaPrefix:complexContent>\n".$contentStr."  </$schemaPrefix:complexContent>\n";
 480              }
 481              // finalize complex type

 482              if($contentStr != ''){
 483                  $contentStr = " <$schemaPrefix:complexType name=\"$typeName\">\n".$contentStr." </$schemaPrefix:complexType>\n";
 484              } else {
 485                  $contentStr = " <$schemaPrefix:complexType name=\"$typeName\"/>\n";
 486              }
 487              $xml .= $contentStr;
 488          }
 489          // simple types

 490          if(isset($this->simpleTypes) && count($this->simpleTypes) > 0){
 491              foreach($this->simpleTypes as $typeName => $attr){
 492                  $xml .= " <$schemaPrefix:simpleType name=\"$typeName\">\n  <restriction base=\"".$this->contractQName($eParts['type'])."\"/>\n </$schemaPrefix:simpleType>";
 493              }
 494          }
 495          // elements

 496          if(isset($this->elements) && count($this->elements) > 0){
 497              foreach($this->elements as $element => $eParts){
 498                  $xml .= " <$schemaPrefix:element name=\"$element\" type=\"".$this->contractQName($eParts['type'])."\"/>\n";
 499              }
 500          }
 501          // attributes

 502          if(isset($this->attributes) && count($this->attributes) > 0){
 503              foreach($this->attributes as $attr => $aParts){
 504                  $xml .= " <$schemaPrefix:attribute name=\"$attr\" type=\"".$this->contractQName($aParts['type'])."\"\n/>";
 505              }
 506          }
 507          // finish 'er up

 508          $el = "<$schemaPrefix:schema targetNamespace=\"$this->schemaTargetNamespace\"\n";
 509          foreach (array_diff($this->usedNamespaces, $this->enclosingNamespaces) as $nsp => $ns) {
 510              $el .= " xmlns:$nsp=\"$ns\"\n";
 511          }
 512          $xml = $el . ">\n".$xml."</$schemaPrefix:schema>\n";
 513          return $xml;
 514      }

 548      /**

 549      * returns an array of information about a given type

 550      * returns false if no type exists by the given name

 551      *

 552      *     typeDef = array(

 553      *     'elements' => array(), // refs to elements array

 554      *    'restrictionBase' => '',

 555      *    'phpType' => '',

 556      *    'order' => '(sequence|all)',

 557      *    'attrs' => array() // refs to attributes array

 558      *    )

 559      *

 560      * @param string

 561      * @return mixed

 562      * @access public

 563      */
 564      function getTypeDef($type){
 565          //$this->debug("in getTypeDef for type $type");

 566          if(isset($this->complexTypes[$type])){
 567              $this->xdebug("in getTypeDef, found complexType $type");
 568              return $this->complexTypes[$type];
 569          } elseif(isset($this->simpleTypes[$type])){
 570              $this->xdebug("in getTypeDef, found simpleType $type");
 571              if (!isset($this->simpleTypes[$type]['phpType'])) {
 572                  // get info for type to tack onto the simple type

 573                  // TODO: can this ever really apply (i.e. what is a simpleType really?)

 574                  $uqType = substr($this->simpleTypes[$type]['type'], strrpos($this->simpleTypes[$type]['type'], ':') + 1);
 575                  $ns = substr($this->simpleTypes[$type]['type'], 0, strrpos($this->simpleTypes[$type]['type'], ':'));
 576                  $etype = $this->getTypeDef($uqType);
 577                  if ($etype) {
 578                      if (isset($etype['phpType'])) {
 579                          $this->simpleTypes[$type]['phpType'] = $etype['phpType'];
 580                      }
 581                      if (isset($etype['elements'])) {
 582                          $this->simpleTypes[$type]['elements'] = $etype['elements'];
 583                      }
 584                  }
 585              }
 586              return $this->simpleTypes[$type];
 587          } elseif(isset($this->elements[$type])){
 588              $this->xdebug("in getTypeDef, found element $type");
 589              if (!isset($this->elements[$type]['phpType'])) {
 590                  // get info for type to tack onto the element
