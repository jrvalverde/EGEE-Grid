 103      function service($data){
 104          global $QUERY_STRING;
 105          if(isset($_SERVER['QUERY_STRING'])){
 106              $qs = $_SERVER['QUERY_STRING'];
 107          } elseif(isset($GLOBALS['QUERY_STRING'])){
 108              $qs = $GLOBALS['QUERY_STRING'];
 109          } elseif(isset($QUERY_STRING) && $QUERY_STRING != ''){
 110              $qs = $QUERY_STRING;
 111          }
 112  
 113          if(isset($qs) && ereg('wsdl', $qs) ){
 114              // This is a request for WSDL

 115              if($this->externalWSDLURL){
 116                if (strpos($this->externalWSDLURL,"://")!==false) { // assume URL
 117                  header('Location: '.$this->externalWSDLURL);
 118                } else { // assume file
 119                  header("Content-Type: text/xml\r\n");
 120                  $fp = fopen($this->externalWSDLURL, 'r');
 121                  fpassthru($fp);
 122                }
 123              } else {
 124                  header("Content-Type: text/xml; charset=ISO-8859-1\r\n");
 125                  print $this->wsdl->serialize();
 126              }
 127          } elseif($data == '' && $this->wsdl){
 128              // print web interface

 129              print $this->webDescription();
 130          } else {
 131              // handle the request

 132              $this->parse_request($data);
 133              if (! $this->fault) {
 134                  $this->invoke_method();
 135              }
 136              if (! $this->fault) {
 137                  $this->serialize_return();
 138              }
 139              $this->send_response();
 140          }

 271      function parse_request($data='') {
 272          $this->debug('entering parse_request() on '.date('H:i Y-m-d'));
 273          $this->parse_http_headers();
 274          $this->debug('got character encoding: '.$this->xml_encoding);
 275          // uncompress if necessary

 276          if (isset($this->headers['Content-Encoding']) && $this->headers['Content-Encoding'] != '') {
 277              $this->debug('got content encoding: ' . $this->headers['Content-Encoding']);
 278              if ($this->headers['Content-Encoding'] == 'deflate' || $this->headers['Content-Encoding'] == 'gzip') {
 279                  // if decoding works, use it. else assume data wasn't gzencoded

 280                  if (function_exists('gzuncompress')) {
 281                      if ($this->headers['Content-Encoding'] == 'deflate' && $degzdata = @gzuncompress($data)) {
 282                          $data = $degzdata;
 283                      } elseif ($this->headers['Content-Encoding'] == 'gzip' && $degzdata = gzinflate(substr($data, 10))) {
 284                          $data = $degzdata;
 285                      } else {
 286                          $this->fault('Server', 'Errors occurred when trying to decode the data');
 287                          return;
 288                      }
 289                  } else {
 290                      $this->fault('Server', 'This Server does not support compressed data');
 291                      return;
 292                  }
 293              }
 294          }
 295          $this->request .= "\r\n".$data;
 296          $this->requestSOAP = $data;
 297          // parse response, get soap parser obj

 298          $parser = new soap_parser($data,$this->xml_encoding);
 299          // parser debug

 300          $this->debug("parser debug: \n".$parser->debug_str);
 301          // if fault occurred during message parsing

 302          if($err = $parser->getError()){
 303              $this->result = 'fault: error in msg parsing: '.$err;
 304              $this->fault('Server',"error in msg parsing:\n".$err);
 305          // else successfully parsed request into soapval object

 306          } else {
 307              // get/set methodname

 308              $this->methodURI = $parser->root_struct_namespace;
 309              $this->methodname = $parser->root_struct_name;
 310              $this->debug('method name: '.$this->methodname);
 311              $this->debug('calling parser->get_response()');
 312              $this->methodparams = $parser->get_response();
 313              // get SOAP headers

 314              $this->requestHeaders = $parser->getHeaders();
 315              // add document for doclit support

 316              $this->document = $parser->document;
 317          }
 318          $this->debug('leaving parse_request() on '.date('H:i Y-m-d'));
 319      }

 338      function invoke_method() {
 339          $this->debug('entering invoke_method');
 340          // does method exist?

 341          if(!function_exists($this->methodname)){
 342              // "method not found" fault here

 343              $this->debug("method '$this->methodname' not found!");
 344              $this->result = 'fault: method not found';
 345              $this->fault('Server',"method '$this->methodname' not defined in service");
 346              return;
 347          }
 348          if($this->wsdl){
 349              if(!$this->opData = $this->wsdl->getOperationData($this->methodname)){
 350              //if(

 351                  $this->fault('Server',"Operation '$this->methodname' is not defined in the WSDL for this service");
 352                  return;
 353              }
 354              $this->debug('opData is ' . $this->varDump($this->opData));
 355          }
 356          $this->debug("method '$this->methodname' exists");
 357          // evaluate message, getting back parameters

 358          // verify that request parameters match the method's signature

 359          if(! $this->verify_method($this->methodname,$this->methodparams)){
 360              // debug

 361              $this->debug('ERROR: request not verified against method signature');
 362              $this->result = 'fault: request failed validation against method signature';
 363              // return fault

 364              $this->fault('Server',"Operation '$this->methodname' not defined in service.");
 365              return;
 366          }
 367  
 368          // if there are parameters to pass

 369          $this->debug('params var dump '.$this->varDump($this->methodparams));
 370          if($this->methodparams){
 371              $this->debug("calling '$this->methodname' with params");
 372              if (! function_exists('call_user_func_array')) {
 373                  $this->debug('calling method using eval()');
 374                  $funcCall = $this->methodname.'(';
 375                  foreach($this->methodparams as $param) {
 376                      $funcCall .= "\"$param\",";
 377                  }
 378                  $funcCall = substr($funcCall, 0, -1).')';
 379                  $this->debug('function call:<br>'.$funcCall);
 380                  @eval("\$this->methodreturn = $funcCall;");
 381              } else {
 382                  $this->debug('calling method using call_user_func_array()');
 383                  $this->methodreturn = call_user_func_array("$this->methodname",$this->methodparams);
 384              }
 385          } else {
 386              // call method w/ no parameters

 387              $this->debug("calling $this->methodname w/ no params");
 388              $m = $this->methodname;
 389              $this->methodreturn = @$m();
 390          }
 391          $this->debug('methodreturn var dump'.$this->varDump($this->methodreturn));
 392          $this->debug("leaving invoke_method: called method $this->methodname, received $this->methodreturn of type ".gettype($this->methodreturn));
 393      }
