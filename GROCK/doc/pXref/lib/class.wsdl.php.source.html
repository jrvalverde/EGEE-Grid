 164      /**

 165       * parses the wsdl document

 166       * 

 167       * @param string $wsdl path or URL

 168       * @access private 

 169       */
 170      function parseWSDL($wsdl = '')
 171      {
 172          if ($wsdl == '') {
 173              $this->debug('no wsdl passed to parseWSDL()!!');
 174              $this->setError('no wsdl passed to parseWSDL()!!');
 175              return false;
 176          }
 177          
 178          // parse $wsdl for url format

 179          $wsdl_props = parse_url($wsdl);
 180  
 181          if (isset($wsdl_props['scheme']) && ($wsdl_props['scheme'] == 'http' || $wsdl_props['scheme'] == 'https')) {
 182              $this->debug('getting WSDL http(s) URL ' . $wsdl);
 183              // get wsdl

 184              $tr = new soap_transport_http($wsdl);
 185              $tr->request_method = 'GET';
 186              $tr->useSOAPAction = false;
 187              if($this->proxyhost && $this->proxyport){
 188                  $tr->setProxy($this->proxyhost,$this->proxyport,$this->proxyusername,$this->proxypassword);
 189              }
 190              if (isset($wsdl_props['user'])) {
 191                  $tr->setCredentials($wsdl_props['user'],$wsdl_props['pass']);
 192              }
 193              $wsdl_string = $tr->send('', $this->timeout, $this->response_timeout);
 194              //$this->debug("WSDL request\n" . $tr->outgoing_payload);

 195              //$this->debug("WSDL response\n" . $tr->incoming_payload);

 196              $this->debug("transport debug data...\n" . $tr->debug_str);
 197              // catch errors

 198              if($err = $tr->getError() ){
 199                  $errstr = 'HTTP ERROR: '.$err;
 200                  $this->debug($errstr);
 201                  $this->setError($errstr);
 202                  unset($tr);
 203                  return false;
 204              }
 205              unset($tr);
 206          } else {
 207              // $wsdl is not http(s), so treat it as a file URL or plain file path

 208              if (isset($wsdl_props['scheme']) && ($wsdl_props['scheme'] == 'file') && isset($wsdl_props['path'])) {
 209                  $path = isset($wsdl_props['host']) ? ($wsdl_props['host'] . ':' . $wsdl_props['path']) : $wsdl_props['path'];
 210              } else {
 211                  $path = $wsdl;
 212              }
 213              $this->debug('getting WSDL file ' . $path);
 214              if ($fp = @fopen($path, 'r')) {
 215                  $wsdl_string = '';
 216                  while ($data = fread($fp, 32768)) {
 217                      $wsdl_string .= $data;
 218                  } 
 219                  fclose($fp);
 220              } else {
 221                  $errstr = "Bad path to WSDL file $path";
 222                  $this->debug($errstr);
 223                  $this->setError($errstr);
 224                  return false;
 225              } 
 226          }
 227          // end new code added

 228          // Create an XML parser.

 229          $this->parser = xml_parser_create(); 
 230          // Set the options for parsing the XML data.

 231          // xml_parser_set_option($parser, XML_OPTION_SKIP_WHITE, 1);

 232          xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING, 0); 
 233          // Set the object for the parser.

 234          xml_set_object($this->parser, $this); 
 235          // Set the element handlers for the parser.

 236          xml_set_element_handler($this->parser, 'start_element', 'end_element');
 237          xml_set_character_data_handler($this->parser, 'character_data');
 238          // Parse the XML file.

 239          if (!xml_parse($this->parser, $wsdl_string, true)) {
 240              // Display an error message.

 241              $errstr = sprintf(
 242                  'XML error parsing WSDL from %s on line %d: %s',
 243                  $wsdl,
 244                  xml_get_current_line_number($this->parser),
 245                  xml_error_string(xml_get_error_code($this->parser))
 246                  );
 247              $this->debug($errstr);
 248              $this->debug("XML payload:\n" . $wsdl_string);
 249              $this->setError($errstr);
 250              return false;
 251          } 
 252          // free the parser

 253          xml_parser_free($this->parser);
 254          // catch wsdl parse errors

 255          if($this->getError()){
 256              return false;
 257          }
 258          return true;
 259      }

 262       * start-element handler

 263       * 

 264       * @param string $parser XML parser object

 265       * @param string $name element name

 266       * @param string $attrs associative array of attributes

 267       * @access private

 269      function start_element($parser, $name, $attrs)
 270      {
 271          if ($this->status == 'schema') {
 272              $this->currentSchema->schemaStartElement($parser, $name, $attrs);
 273              $this->debug_str .= $this->currentSchema->debug_str;
 274              $this->currentSchema->debug_str = '';
 275          } elseif (ereg('schema$', $name)) {
 276              // $this->debug("startElement for $name ($attrs[name]). status = $this->status (".$this->getLocalPart($name).")");

 277              $this->status = 'schema';
 278              $this->currentSchema = new xmlschema('', '', $this->namespaces);
 279              $this->currentSchema->schemaStartElement($parser, $name, $attrs);
 280              $this->debug_str .= $this->currentSchema->debug_str;
 281              $this->currentSchema->debug_str = '';
 282          } else {
 283              // position in the total number of elements, starting from 0

 284              $pos = $this->position++;
 285              $depth = $this->depth++; 
 286              // set self as current value for this depth

 287              $this->depth_array[$depth] = $pos;
 288              $this->message[$pos] = array('cdata' => ''); 
 289              // get element prefix

 290              if (ereg(':', $name)) {
 291                  // get ns prefix

 292                  $prefix = substr($name, 0, strpos($name, ':')); 
 293                  // get ns

 294                  $namespace = isset($this->namespaces[$prefix]) ? $this->namespaces[$prefix] : ''; 
 295                  // get unqualified name

 296                  $name = substr(strstr($name, ':'), 1);
 297              } 
 298  
 299              if (count($attrs) > 0) {
 300                  foreach($attrs as $k => $v) {
 301                      // if ns declarations, add to class level array of valid namespaces

 302                      if (ereg("^xmlns", $k)) {
 303                          if ($ns_prefix = substr(strrchr($k, ':'), 1)) {
 304                              $this->namespaces[$ns_prefix] = $v;
 305                          } else {
 306                              $this->namespaces['ns' . (count($this->namespaces) + 1)] = $v;
 307                          } 
 308                          if ($v == 'http://www.w3.org/2001/XMLSchema' || $v == 'http://www.w3.org/1999/XMLSchema') {
 309                              $this->XMLSchemaVersion = $v;
 310                              $this->namespaces['xsi'] = $v . '-instance';
 311                          } 
 312                      } //  
 313                      // expand each attribute

 314                      $k = strpos($k, ':') ? $this->expandQname($k) : $k;
 315                      if ($k != 'location' && $k != 'soapAction' && $k != 'namespace') {
 316                          $v = strpos($v, ':') ? $this->expandQname($v) : $v;
 317                      } 
 318                      $eAttrs[$k] = $v;
 319                  } 
 320                  $attrs = $eAttrs;
 321              } else {
 322                  $attrs = array();
 323              } 
 324              // find status, register data

 325              switch ($this->status) {
 326                  case 'message':
 327                      if ($name == 'part') {
 328                          if (isset($attrs['type'])) {
 329                              $this->debug("msg " . $this->currentMessage . ": found part $attrs[name]: " . implode(',', $attrs));
 330                              $this->messages[$this->currentMessage][$attrs['name']] = $attrs['type'];
 331                          } 
 332                          if (isset($attrs['element'])) {
 333                              $this->messages[$this->currentMessage][$attrs['name']] = $attrs['element'];
 334                          } 
 335                      } 
 336                      break;
 337                  case 'portType':
 338                      switch ($name) {
 339                          case 'operation':
 340                              $this->currentPortOperation = $attrs['name'];
 341                              $this->debug("portType $this->currentPortType operation: $this->currentPortOperation");
 342                              if (isset($attrs['parameterOrder'])) {
 343                                  $this->portTypes[$this->currentPortType][$attrs['name']]['parameterOrder'] = $attrs['parameterOrder'];
 344                              } 
 345                              break;
 346                          case 'documentation':
 347                              $this->documentation = true;
 348                              break; 
 349                          // merge input/output data

 350                          default:
 351                              $m = isset($attrs['message']) ? $this->getLocalPart($attrs['message']) : '';
 352                              $this->portTypes[$this->currentPortType][$this->currentPortOperation][$name]['message'] = $m;
 353                              break;
 354                      } 
 355                      break;
 356                  case 'binding':
 357                      switch ($name) {
 358                          case 'binding': 
 359                              // get ns prefix

 360                              if (isset($attrs['style'])) {
 361                              $this->bindings[$this->currentBinding]['prefix'] = $prefix;
 362                              } 
 363                              $this->bindings[$this->currentBinding] = array_merge($this->bindings[$this->currentBinding], $attrs);
 364                              break;
 365                          case 'header':
 366                              $this->bindings[$this->currentBinding]['operations'][$this->currentOperation][$this->opStatus]['headers'][] = $attrs;
 367                              break;
 368                          case 'operation':
 369                              if (isset($attrs['soapAction'])) {
 370                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation]['soapAction'] = $attrs['soapAction'];
 371                              } 
 372                              if (isset($attrs['style'])) {
 373                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation]['style'] = $attrs['style'];
 374                              } 
 375                              if (isset($attrs['name'])) {
 376                                  $this->currentOperation = $attrs['name'];
 377                                  $this->debug("current binding operation: $this->currentOperation");
 378                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation]['name'] = $attrs['name'];
 379                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation]['binding'] = $this->currentBinding;
 380                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation]['endpoint'] = isset($this->bindings[$this->currentBinding]['endpoint']) ? $this->bindings[$this->currentBinding]['endpoint'] : '';
 381                              } 
 382                              break;
 383                          case 'input':
 384                              $this->opStatus = 'input';
 385                              break;
 386                          case 'output':
 387                              $this->opStatus = 'output';
 388                              break;
 389                          case 'body':
 390                              if (isset($this->bindings[$this->currentBinding]['operations'][$this->currentOperation][$this->opStatus])) {
 391                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation][$this->opStatus] = array_merge($this->bindings[$this->currentBinding]['operations'][$this->currentOperation][$this->opStatus], $attrs);
 392                              } else {
 393                                  $this->bindings[$this->currentBinding]['operations'][$this->currentOperation][$this->opStatus] = $attrs;
 394                              } 
 395                              break;
 396                      } 
 397                      break;
 398                  case 'service':
 399                      switch ($name) {
 400                          case 'port':
 401                              $this->currentPort = $attrs['name'];
 402                              $this->debug('current port: ' . $this->currentPort);
 403                              $this->ports[$this->currentPort]['binding'] = $this->getLocalPart($attrs['binding']);
 404                      
 405                              break;
 406                          case 'address':
 407                              $this->ports[$this->currentPort]['location'] = $attrs['location'];
 408                              $this->ports[$this->currentPort]['bindingType'] = $namespace;
 409                              $this->bindings[ $this->ports[$this->currentPort]['binding'] ]['bindingType'] = $namespace;
 410                              $this->bindings[ $this->ports[$this->currentPort]['binding'] ]['endpoint'] = $attrs['location'];
 411                              break;
 412                      } 
 413                      break;
 414              } 
 415          // set status

 416          switch ($name) {
 417              case 'import':
 418                  if (isset($attrs['location'])) {
 419                      $this->import[$attrs['namespace']][] = array('location' => $attrs['location'], 'loaded' => false);
 420                      $this->debug('parsing import ' . $attrs['namespace']. ' - ' . $attrs['location'] . ' (' . count($this->import[$attrs['namespace']]).')');
 421                  } else {
 422                      $this->import[$attrs['namespace']][] = array('location' => '', 'loaded' => true);
 423                      if (! $this->getPrefixFromNamespace($attrs['namespace'])) {
 424                          $this->namespaces['ns'.(count($this->namespaces)+1)] = $attrs['namespace'];
 425                      }
 426                      $this->debug('parsing import ' . $attrs['namespace']. ' - [no location] (' . count($this->import[$attrs['namespace']]).')');
 427                  }
 428                  break;
 429              //wait for schema

 430              //case 'types':

 431              //    $this->status = 'schema';

 432              //    break;

 433              case 'message':
 434                  $this->status = 'message';
 435                  $this->messages[$attrs['name']] = array();
 436                  $this->currentMessage = $attrs['name'];
 437                  break;
 438              case 'portType':
 439                  $this->status = 'portType';
 440                  $this->portTypes[$attrs['name']] = array();
 441                  $this->currentPortType = $attrs['name'];
 442                  break;
 443              case "binding":
 444                  if (isset($attrs['name'])) {
 445                  // get binding name

 446                      if (strpos($attrs['name'], ':')) {
 447                          $this->currentBinding = $this->getLocalPart($attrs['name']);
 448                      } else {
 449                          $this->currentBinding = $attrs['name'];
 450                      } 
 451                      $this->status = 'binding';
 452                      $this->bindings[$this->currentBinding]['portType'] = $this->getLocalPart($attrs['type']);
 453                      $this->debug("current binding: $this->currentBinding of portType: " . $attrs['type']);
 454                  } 
 455                  break;
 456              case 'service':
 457                  $this->serviceName = $attrs['name'];
 458                  $this->status = 'service';
 459                  $this->debug('current service: ' . $this->serviceName);
 460                  break;
 461              case 'definitions':
 462                  foreach ($attrs as $name => $value) {
 463                      $this->wsdl_info[$name] = $value;
 464                  } 
 465                  break;
 466              } 
 467          } 
 468      }

 477      function end_element($parser, $name){ 
 478          // unset schema status

 479          if (/*ereg('types$', $name) ||*/ ereg('schema$', $name)) {
 480              $this->status = "";
 481              $this->schemas[$this->currentSchema->schemaTargetNamespace][] = $this->currentSchema;
 482          } 
 483          if ($this->status == 'schema') {
 484              $this->currentSchema->schemaEndElement($parser, $name);
 485          } else {
 486              // bring depth down a notch

 487              $this->depth--;
 488          } 
 489          // end documentation

 490          if ($this->documentation) {
 491              //TODO: track the node to which documentation should be assigned; it can be a part, message, etc.

 492              //$this->portTypes[$this->currentPortType][$this->currentPortOperation]['documentation'] = $this->documentation;

 493              $this->documentation = false;
 494          } 
 495      }

 583      *

 584      *     typeDef = array(

 585      *     'elements' => array(), // refs to elements array

 586      *    'restrictionBase' => '',

 587      *    'phpType' => '',

 588      *    'order' => '(sequence|all)',

 589      *    'attrs' => array() // refs to attributes array

 590      *    )

 591      *

 592      * @param $type string

 593      * @param $ns string

 594      * @return mixed

 595      * @access public

 596      * @see xmlschema

 597      */
 598      function getTypeDef($type, $ns) {
 599          if ((! $ns) && isset($this->namespaces['tns'])) {
 600              $ns = $this->namespaces['tns'];
 601          }
 602          if (isset($this->schemas[$ns])) {
 603              foreach ($this->schemas[$ns] as $xs) {
 604                  $t = $xs->getTypeDef($type);
 605                  $this->debug_str .= $xs->debug_str;
 606                  $xs->debug_str = '';
 607                  if ($t) {
 608                      return $t;
 609                  }
 610              }
 611          }
 612          return false;
 613      }

 615      /**

 616      * serialize the parsed wsdl

 617      * 

 618      * @return string , serialization of WSDL

 619      * @access public 

 620      */
 621      function serialize()
 622      {
 623          $xml = '<?xml version="1.0" encoding="ISO-8859-1"?><definitions';
 624          foreach($this->namespaces as $k => $v) {
 625              $xml .= " xmlns:$k=\"$v\"";
 626          } 
 627          // 10.9.02 - add poulter fix for wsdl and tns declarations

 628          if (isset($this->namespaces['wsdl'])) {
 629              $xml .= " xmlns=\"" . $this->namespaces['wsdl'] . "\"";
 630          } 
 631          if (isset($this->namespaces['tns'])) {
 632              $xml .= " targetNamespace=\"" . $this->namespaces['tns'] . "\"";
 633          } 
 634          $xml .= '>'; 
 635          // imports

 636          if (sizeof($this->import) > 0) {
 637              foreach($this->import as $ns => $list) {
 638                  foreach ($list as $ii) {
 639                      if ($ii['location'] != '') {
 640                          $xml .= '<import location="' . $ii['location'] . '" namespace="' . $ns . '" />';
 641                      } else {
 642                          $xml .= '<import namespace="' . $ns . '" />';
 643                      }
 644                  }
 645              } 
 646          } 
 647          // types

 648          if (count($this->schemas)>=1) {
 649              $xml .= '<types>';
 650              foreach ($this->schemas as $ns => $list) {
 651                  foreach ($list as $xs) {
 652                      $xml .= $xs->serializeSchema();
 653                  }
 654              }
 655              $xml .= '</types>';
 656          } 
 657          // messages

 658          if (count($this->messages) >= 1) {
 659              foreach($this->messages as $msgName => $msgParts) {
 660                  $xml .= '<message name="' . $msgName . '">';
 661                  if(is_array($msgParts)){
 662                      foreach($msgParts as $partName => $partType) {
 663                          // print 'serializing '.$partType.', sv: '.$this->XMLSchemaVersion.'<br>';

 664                          if (strpos($partType, ':')) {
 665                              $typePrefix = $this->getPrefixFromNamespace($this->getPrefix($partType));
 666                          } elseif (isset($this->typemap[$this->namespaces['xsd']][$partType])) {
 667                              // print 'checking typemap: '.$this->XMLSchemaVersion.'<br>';

 668                              $typePrefix = 'xsd';
 669                          } else {
 670                              foreach($this->typemap as $ns => $types) {
 671                                  if (isset($types[$partType])) {
 672                                      $typePrefix = $this->getPrefixFromNamespace($ns);
 673                                  } 
 674                              } 
 675                              if (!isset($typePrefix)) {
 676                                  die("$partType has no namespace!");
 677                              } 
 678                          } 
 679                          $xml .= '<part name="' . $partName . '" type="' . $typePrefix . ':' . $this->getLocalPart($partType) . '" />';
 680                      }
 681                  }
 682                  $xml .= '</message>';
 683              } 
 684          } 
 685          // bindings & porttypes

 686          if (count($this->bindings) >= 1) {
 687              $binding_xml = '';
 688              $portType_xml = '';
 689              foreach($this->bindings as $bindingName => $attrs) {
 690                  $binding_xml .= '<binding name="' . $bindingName . '" type="tns:' . $attrs['portType'] . '">';
 691                  $binding_xml .= '<soap:binding style="' . $attrs['style'] . '" transport="' . $attrs['transport'] . '"/>';
 692                  $portType_xml .= '<portType name="' . $attrs['portType'] . '">';
 693                  foreach($attrs['operations'] as $opName => $opParts) {
 694                      $binding_xml .= '<operation name="' . $opName . '">';
 695                      $binding_xml .= '<soap:operation soapAction="' . $opParts['soapAction'] . '" style="'. $attrs['style'] . '"/>';
 696                      if (isset($opParts['input']['encodingStyle']) && $opParts['input']['encodingStyle'] != '') {
 697                          $enc_style = ' encodingStyle="' . $opParts['input']['encodingStyle'] . '"';
 698                      } else {
 699                          $enc_style = '';
 700                      }
 701                      $binding_xml .= '<input><soap:body use="' . $opParts['input']['use'] . '" namespace="' . $opParts['input']['namespace'] . '"' . $enc_style . '/></input>';
 702                      if (isset($opParts['output']['encodingStyle']) && $opParts['output']['encodingStyle'] != '') {
 703                          $enc_style = ' encodingStyle="' . $opParts['output']['encodingStyle'] . '"';
 704                      } else {
 705                          $enc_style = '';
 706                      }
 707                      $binding_xml .= '<output><soap:body use="' . $opParts['output']['use'] . '" namespace="' . $opParts['output']['namespace'] . '"' . $enc_style . '/></output>';
 708                      $binding_xml .= '</operation>';
 709                      $portType_xml .= '<operation name="' . $opParts['name'] . '"';
 710                      if (isset($opParts['parameterOrder'])) {
 711                          $portType_xml .= ' parameterOrder="' . $opParts['parameterOrder'] . '"';
 712                      } 
 713                      $portType_xml .= '>';
 714                      if(isset($opParts['documentation']) && $opParts['documentation'] != '') {
 715                          $portType_xml .= '<documentation>' . htmlspecialchars($opParts['documentation']) . '</documentation>';
 716                      }
 717                      $portType_xml .= '<input message="tns:' . $opParts['input']['message'] . '"/>';
 718                      $portType_xml .= '<output message="tns:' . $opParts['output']['message'] . '"/>';
 719                      $portType_xml .= '</operation>';
 720                  } 
 721                  $portType_xml .= '</portType>';
 722                  $binding_xml .= '</binding>';
 723              } 
 724              $xml .= $portType_xml . $binding_xml;
 725          } 
 726          // services

 727          $xml .= '<service name="' . $this->serviceName . '">';
 728          if (count($this->ports) >= 1) {
 729              foreach($this->ports as $pName => $attrs) {
 730                  $xml .= '<port name="' . $pName . '" binding="tns:' . $attrs['binding'] . '">';
 731                  $xml .= '<soap:address location="' . $attrs['location'] . '"/>';
 732                  $xml .= '</port>';
 733              } 
 734          } 
 735          $xml .= '</service>';
 736          return $xml . '</definitions>';
 737      }

 889      /**

 890       * serializes a PHP value according a given type definition

 891       * 

 892       * @param string $name , name of type (part)

 893       * @param string $type , type of type, heh (type or element)

 894       * @param mixed $value , a native PHP value (parameter value)

 895       * @param string $use , use for part (encoded|literal)

 896       * @param string $encodingStyle , use to add encoding changes to serialisation

 897       * @return string serialization

 898       * @access public 

 899       */
 900      function serializeType($name, $type, $value, $use='encoded', $encodingStyle=false)
 901      {
 902          $this->debug("in serializeType: $name, $type, $value, $use, $encodingStyle");
 903          if($use == 'encoded' && $encodingStyle) {
 904              $encodingStyle = ' SOAP-ENV:encodingStyle="' . $encodingStyle . '"';
 905          }
 906  
 907          // if a soap_val has been supplied, let its type override the WSDL

 908          if (is_object($value) && get_class($value) == 'soapval') {
 909              // TODO: get attributes from soapval?

 910              if ($value->type_ns) {
 911                  $type = $value->type_ns . ':' . $value->type;
 912              } else {
 913                  $type = $value->type;
 914              }
 915              $value = $value->value;
 916              $forceType = true;
 917              $this->debug("in serializeType: soapval overrides type to $type, value to $value");
 918          } else {
 919              $forceType = false;
 920          }
 921  
 922          $xml = '';
 923          if (strpos($type, ':')) {
 924              $uqType = substr($type, strrpos($type, ':') + 1);
 925              $ns = substr($type, 0, strrpos($type, ':'));
 926              $this->debug("got a prefixed type: $uqType, $ns");
 927              if ($this->getNamespaceFromPrefix($ns)) {
 928                  $ns = $this->getNamespaceFromPrefix($ns);
 929                  $this->debug("expanded prefixed type: $uqType, $ns");
 930              }
 931  
 932              if($ns == $this->XMLSchemaVersion){
 933                  
 934                  if (is_null($value)) {
 935                      if ($use == 'literal') {
 936                          // TODO: depends on nillable

 937                          return "<$name/>";
 938                      } else {
 939                          return "<$name xsi:nil=\"true\"/>";
 940                      }
 941                  }
 942                  if ($uqType == 'boolean' && !$value) {
 943                      $value = 'false';
 944                  } elseif ($uqType == 'boolean') {
 945                      $value = 'true';
 946                  } 
 947                  if ($uqType == 'string' && gettype($value) == 'string') {
 948                      $value = $this->expandEntities($value);
 949                  } 
 950                  // it's a scalar

 951                  // TODO: what about null/nil values?

 952                  // check type isn't a custom type extending xmlschema namespace

 953                  if (!$this->getTypeDef($uqType, $ns)) {
 954                      if ($use == 'literal') {
 955                          if ($forceType) {
 956                              return "<$name xsi:type=\"" . $this->getPrefixFromNamespace($this->XMLSchemaVersion) . ":$uqType\">$value</$name>";
 957                          } else {
 958                              return "<$name>$value</$name>";
 959                          }
 960                      } else {
 961                          return "<$name xsi:type=\"" . $this->getPrefixFromNamespace($this->XMLSchemaVersion) . ":$uqType\"$encodingStyle>$value</$name>";
 962                      }
 963                  }
 964              } else if ($ns == 'http://xml.apache.org/xml-soap') {
 965                  if ($uqType == 'Map') {
 966                      $contents = '';
 967                      foreach($value as $k => $v) {
 968                          $this->debug("serializing map element: key $k, value $v");
 969                          $contents .= '<item>';
 970                          $contents .= $this->serialize_val($k,'key',false,false,false,false,$use);
 971                          $contents .= $this->serialize_val($v,'value',false,false,false,false,$use);
 972                          $contents .= '</item>';
 973                      }
 974                      if ($use == 'literal') {
 975                          if ($forceType) {
 976                          return "<$name xsi:type=\"" . $this->getPrefixFromNamespace('http://xml.apache.org/xml-soap') . ":$uqType\">$contents</$name>";
 977                          } else {
 978                              return "<$name>$contents</$name>";
 979                          }
 980                      } else {
 981                          return "<$name xsi:type=\"" . $this->getPrefixFromNamespace('http://xml.apache.org/xml-soap') . ":$uqType\"$encodingStyle>$contents</$name>";
 982                      }
 983                  }
 984              }
 985          } else {
 986              $this->debug("No namespace for type $type");
 987              $ns = '';
 988              $uqType = $type;
 989          }
 990          if(!$typeDef = $this->getTypeDef($uqType, $ns)){
 991              $this->setError("$type ($uqType) is not a supported type.");
 992              $this->debug("$type ($uqType) is not a supported type.");
 993              return false;
 994          } else {
 995              foreach($typeDef as $k => $v) {
 996                  $this->debug("typedef, $k: $v");
 997              }
 998          }
 999          $phpType = $typeDef['phpType'];
1000          $this->debug("serializeType: uqType: $uqType, ns: $ns, phptype: $phpType, arrayType: " . (isset($typeDef['arrayType']) ? $typeDef['arrayType'] : '') ); 
1001          // if php type == struct, map value to the <all> element names

1002          if ($phpType == 'struct') {
1003              if (isset($typeDef['typeClass']) && $typeDef['typeClass'] == 'element') {
1004                  $elementName = $uqType;
1005                  if (isset($typeDef['form']) && ($typeDef['form'] == 'qualified')) {
1006                      $elementNS = " xmlns=\"$ns\"";
1007                  }
1008              } else {
1009                  $elementName = $name;
1010                  $elementNS = '';
1011              }
1012              if (is_null($value)) {
1013                  if ($use == 'literal') {
1014                      // TODO: depends on nillable

1015                      return "<$elementName$elementNS/>";
1016                  } else {
1017                      return "<$elementName$elementNS xsi:nil=\"true\"/>";
1018                  }
1019              }
1020              if ($use == 'literal') {
1021                  if ($forceType) {
1022                      $xml = "<$elementName$elementNS xsi:type=\"" . $this->getPrefixFromNamespace($ns) . ":$uqType\">";
1023                  } else {
1024                      $xml = "<$elementName$elementNS>";
1025                  }
1026              } else {
1027                  $xml = "<$elementName$elementNS xsi:type=\"" . $this->getPrefixFromNamespace($ns) . ":$uqType\"$encodingStyle>";
1028              }
1029              
1030              if (isset($typeDef['elements']) && is_array($typeDef['elements'])) {
1031                  if (is_array($value)) {
1032                      $xvalue = $value;
1033                  } elseif (is_object($value)) {
1034                      $xvalue = get_object_vars($value);
1035                  } else {
1036                      $this->debug("value is neither an array nor an object for XML Schema type $ns:$uqType");
1037                      $xvalue = array();
1038                  }
1039                  // toggle whether all elements are present - ideally should validate against schema

1040                  if(count($typeDef['elements']) != count($xvalue)){
1041                      $optionals = true;
1042                  }
1043                  foreach($typeDef['elements'] as $eName => $attrs) {
1044                      // if user took advantage of a minOccurs=0, then only serialize named parameters

1045                      if(isset($optionals) && !isset($xvalue[$eName])){
1046                          // do nothing

1047                      } else {
1048                          // get value

1049                          if (isset($xvalue[$eName])) {
1050                              $v = $xvalue[$eName];
1051                          } else {
1052                              $v = null;
1053                          }
1054                          // TODO: if maxOccurs > 1 (not just unbounded), then allow serialization of an array

1055                          if (isset($attrs['maxOccurs']) && $attrs['maxOccurs'] == 'unbounded' && isset($v) && is_array($v) && $this->isArraySimpleOrStruct($v) == 'arraySimple') {
1056                              $vv = $v;
1057                              foreach ($vv as $k => $v) {
1058                                  if (isset($attrs['type'])) {
1059                                      // serialize schema-defined type

1060                                      $xml .= $this->serializeType($eName, $attrs['type'], $v, $use, $encodingStyle);
1061                                  } else {
1062                                      // serialize generic type

1063                                      $this->debug("calling serialize_val() for $v, $eName, false, false, false, false, $use");
1064                                      $xml .= $this->serialize_val($v, $eName, false, false, false, false, $use);
1065                                  }
1066                              }
1067                          } else {
1068                              if (isset($attrs['type'])) {
1069                                  // serialize schema-defined type

1070                                  $xml .= $this->serializeType($eName, $attrs['type'], $v, $use, $encodingStyle);
1071                              } else {
1072                                  // serialize generic type

1073                                  $this->debug("calling serialize_val() for $v, $eName, false, false, false, false, $use");
1074                                  $xml .= $this->serialize_val($v, $eName, false, false, false, false, $use);
1075                              }
1076                          }
1077                      }
1078                  } 
1079              } else {
1080                  $this->debug("Expected elements for XML Schema type $ns:$uqType");
1081              }
1082              $xml .= "</$elementName>";
1083          } elseif ($phpType == 'array') {
1084              if (isset($typeDef['form']) && ($typeDef['form'] == 'qualified')) {
1085                  $elementNS = " xmlns=\"$ns\"";
1086              } else {
1087                  $elementNS = '';
1088              }
1089              if (is_null($value)) {
1090                  if ($use == 'literal') {
1091                      // TODO: depends on nillable

1092                      return "<$name$elementNS/>";
1093                  } else {
1094                      return "<$name$elementNS xsi:nil=\"true\"/>";
1095                  }
1096              }
1097              if (isset($typeDef['multidimensional'])) {
1098                  $nv = array();
1099                  foreach($value as $v) {
1100                      $cols = ',' . sizeof($v);
1101                      $nv = array_merge($nv, $v);
1102                  } 
1103                  $value = $nv;
1104              } else {
1105                  $cols = '';
1106              } 
1107              if (is_array($value) && sizeof($value) >= 1) {
1108                  $rows = sizeof($value);
1109                  $contents = '';
1110                  foreach($value as $k => $v) {
1111                      $this->debug("serializing array element: $k, $v of type: $typeDef[arrayType]");
1112                      //if (strpos($typeDef['arrayType'], ':') ) {

1113                      if (!in_array($typeDef['arrayType'],$this->typemap['http://www.w3.org/2001/XMLSchema'])) {
1114                          $contents .= $this->serializeType('item', $typeDef['arrayType'], $v, $use);
1115                      } else {
1116                          $contents .= $this->serialize_val($v, 'item', $typeDef['arrayType'], null, $this->XMLSchemaVersion, false, $use);
