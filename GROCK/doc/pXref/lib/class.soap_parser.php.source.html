  59          $this->xml = $xml;
  60          $this->xml_encoding = $encoding;
  61          $this->method = $method;
  62          $this->decode_utf8 = $decode_utf8;
  63  
  64          // Check whether content has been read.

  65          if(!empty($xml)){
  66              $this->debug('Entering soap_parser(), length='.strlen($xml).', encoding='.$encoding);
  67              // Create an XML parser - why not xml_parser_create_ns?

  68              $this->parser = xml_parser_create($this->xml_encoding);
  69              // Set the options for parsing the XML data.

  70              //xml_parser_set_option($parser, XML_OPTION_SKIP_WHITE, 1);

  71              xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING, 0);
  72              xml_parser_set_option($this->parser, XML_OPTION_TARGET_ENCODING, $this->xml_encoding);

  77              xml_set_character_data_handler($this->parser,'character_data');

  83                  xml_get_current_line_number($this->parser),

  85                  $this->debug($err);

  88              } else {
  89                  $this->debug('parsed successfully, found root struct: '.$this->root_struct.' of name '.$this->root_struct_name);
  90                  // get final value

  91                  $this->soapresponse = $this->message[$this->root_struct]['result'];
  92                  // get header value: no, because this is documented as XML string

 123      function start_element($parser, $name, $attrs) {
 124          // position in a total number of elements, starting from 0

 125          // update class level pos

 126          $pos = $this->position++;
 127          // and set mine

 128          $this->message[$pos] = array('pos' => $pos,'children'=>'','cdata'=>'');
 129          // depth = how many levels removed from root?

 130          // set mine as current global depth and increment global depth value

 131          $this->message[$pos]['depth'] = $this->depth++;
 132  
 133          // else add self as child to whoever the current parent is

 136          }
 137          // set my parent

 138          $this->message[$pos]['parent'] = $this->parent;
 139          // set self as current parent

 140          $this->parent = $pos;
 141          // set self as current value for this depth

 142          $this->depth_array[$this->depth] = $pos;
 143          // get element prefix

 144          if(strpos($name,':')){
 145              // get ns prefix

 146              $prefix = substr($name,0,strpos($name,':'));
 147              // get unqualified name

 148              $name = substr(strstr($name,':'),1);
 149          }
 150          // set status

 151          if($name == 'Envelope'){
 152              $this->status = 'envelope';
 153          } elseif($name == 'Header'){
 154              $this->root_header = $pos;

 156          } elseif($name == 'Body'){
 157              $this->status = 'body';
 158              $this->body_position = $pos;
 159          // set method

 160          } elseif($this->status == 'body' && $pos == ($this->body_position+1)){

 166          }
 167          // set my status

 168          $this->message[$pos]['status'] = $this->status;
 169          // set name

 170          $this->message[$pos]['name'] = htmlspecialchars($name);
 171          // set attrs

 172          $this->message[$pos]['attrs'] = $attrs;
 173  
 174          // loop through atts, logging ns and type declarations

 175          $attstr = '';
 176          foreach($attrs as $key => $value){
 177              $key_prefix = $this->getPrefix($key);
 178              $key_localpart = $this->getLocalPart($key);
 179              // if ns declarations, add to class level array of valid namespaces

 192              } elseif($key_localpart == 'type'){
 193                  $value_prefix = $this->getPrefix($value);
 194                  $value_localpart = $this->getLocalPart($value);
 195                  $this->message[$pos]['type'] = $value_localpart;
 196                  $this->message[$pos]['typePrefix'] = $value_prefix;
 197                  if(isset($this->namespaces[$value_prefix])){
 198                      $this->message[$pos]['type_namespace'] = $this->namespaces[$value_prefix];
 199                  } else if(isset($attrs['xmlns:'.$value_prefix])) {
 200                      $this->message[$pos]['type_namespace'] = $attrs['xmlns:'.$value_prefix];
 201                  }
 202                  // should do something here with the namespace of specified type?

 203              } elseif($key_localpart == 'arrayType'){
 204                  $this->message[$pos]['type'] = 'array';
 205                  /* do arrayType ereg here

 206                  [1]    arrayTypeValue    ::=    atype asize

 207                  [2]    atype    ::=    QName rank*

 208                  [3]    rank    ::=    '[' (',')* ']'

 225                  }
 226              }
 227              // log id

 228              if($key == 'id'){
 229                  $this->ids[$value] = $pos;
 230              }
 231              // root

 232              if($key_localpart == 'root' && $value == 1){
 233                  $this->status = 'method';
 234                  $this->root_struct_name = $name;
 235                  $this->root_struct = $pos;
 236                  $this->debug("found root struct $this->root_struct_name, pos $pos");
 237              }
 238              // for doclit

 239              $attstr .= " $key=\"$value\"";
 240          }
 241          // get namespace - must be done after namespace atts are processed

 242          if(isset($prefix)){
 243              $this->message[$pos]['namespace'] = $this->namespaces[$prefix];
 244              $this->default_namespace = $this->namespaces[$prefix];
 245          } else {
 246              $this->message[$pos]['namespace'] = $this->default_namespace;
 247          }
 248          if($this->status == 'header'){
 249              if ($this->root_header != $pos) {
 250                  $this->responseHeaders .= "<" . (isset($prefix) ? $prefix . ':' : '') . "$name$attstr>";
 251              }
 252          } elseif($this->root_struct_name != ''){
 253              $this->document .= "<" . (isset($prefix) ? $prefix . ':' : '') . "$name$attstr>";
 254          }

 271              $prefix = substr($name,0,strpos($name,':'));

 273              $name = substr(strstr($name,':'),1);
 274          }
 275          
 276          // build to native type

 277          if(isset($this->body_position) && $pos > $this->body_position){
 278              // deal w/ multirefs

 279              if(isset($this->message[$pos]['attrs']['href'])){
 280                  // get id

 281                  $id = substr($this->message[$pos]['attrs']['href'],1);

 285                  $this->message[$pos]['result'] =& $this->multirefs[$id][$pos];
 286              // build complex values

 287              } elseif($this->message[$pos]['children'] != ''){
 288

 290                  if(!isset($this->message[$pos]['result'])){
 291                      $this->message[$pos]['result'] = $this->buildVal($pos);
 292                  }
 293                  
 294              // set value of simple type

 295              } else {
 296                  //$this->debug('adding data for scalar value '.$this->message[$pos]['name'].' of value '.$this->message[$pos]['cdata']);

 297                  if (isset($this->message[$pos]['type'])) {
 298                      $this->message[$pos]['result'] = $this->decodeSimple($this->message[$pos]['cdata'], $this->message[$pos]['type'], isset($this->message[$pos]['type_namespace']) ? $this->message[$pos]['type_namespace'] : '');

 310                  if($this->message[$parent]['type'] != 'map'){

 311                      if(strtolower($this->message[$parent]['type']) == 'array'){

 312                          $this->message[$parent]['result'][] = $this->message[$pos]['result'];

 313                      } else {

 314                          $this->message[$parent]['result'][$this->message[$pos]['name']] = $this->message[$pos]['result'];

 315                      }

 316                  }

 317                  */
 318              }
 319          }
 320          
 321          // for doclit

 322          if($this->status == 'header'){
 323              if ($this->root_header != $pos) {
 324                  $this->responseHeaders .= "</" . (isset($prefix) ? $prefix . ':' : '') . "$name>";
 325              }
 326          } elseif($pos >= $this->root_struct){

 333          } elseif($name == 'Body'){

 403      /**

 404      * decodes simple types into PHP variables

 405      *

 406      * @param    string $value value to decode

 407      * @param    string $type XML type to decode

 408      * @param    string $typens XML type namespace to decode

 409      * @access   private

 410      */
 411      function decodeSimple($value, $type, $typens) {
 412          // TODO: use the namespace!

 413          if ((!isset($type)) || $type == 'string' || $type == 'long' || $type == 'unsignedLong') {
 414              return (string) $value;
 415          }
 416          if ($type == 'int' || $type == 'integer' || $type == 'short' || $type == 'byte') {
 417              return (int) $value;
 418          }
 419          if ($type == 'float' || $type == 'double' || $type == 'decimal') {
 420              return (double) $value;
 421          }
 422          if ($type == 'boolean') {
 423              if (strtolower($value) == 'false' || strtolower($value) == 'f') {
 424                  return false;
 425              }
 426              return (boolean) $value;
 427          }
 428          if ($type == 'base64' || $type == 'base64Binary') {
 429              return base64_decode($value);
 430          }
 431          // obscure numeric types

 432          if ($type == 'nonPositiveInteger' || $type == 'negativeInteger'
 433              || $type == 'nonNegativeInteger' || $type == 'positiveInteger'
 434              || $type == 'unsignedInt'
 435              || $type == 'unsignedShort' || $type == 'unsignedByte') {
 436              return (int) $value;
 437          }
 438          // everything else

 439          return (string) $value;
 440      }

 442      /**

 443      * builds response structures for compound values (arrays/structs)

 444      *

 445      * @param    string $pos position in node tree

 446      * @access   private

 447      */
 448      function buildVal($pos){
 449          if(!isset($this->message[$pos]['type'])){
 450              $this->message[$pos]['type'] = '';
 451          }
 452          $this->debug('inside buildVal() for '.$this->message[$pos]['name']."(pos $pos) of type ".$this->message[$pos]['type']);
 453          // if there are children...

 454          if($this->message[$pos]['children'] != ''){
 455              $children = explode('|',$this->message[$pos]['children']);
 456              array_shift($children); // knock off empty
 457              // md array

 458              if(isset($this->message[$pos]['arrayCols']) && $this->message[$pos]['arrayCols'] != ''){
 459                  $r=0; // rowcount
 460                  $c=0; // colcount
 461                  foreach($children as $child_pos){
 462                      $this->debug("got an MD array element: $r, $c");
 463                      $params[$r][] = $this->message[$child_pos]['result'];
 464                      $c++;
 465                      if($c == $this->message[$pos]['arrayCols']){
 466                          $c = 0;
 467                          $r++;
 468                      }
 469                  }
 470              // array

 471              } elseif($this->message[$pos]['type'] == 'array' || $this->message[$pos]['type'] == 'Array'){
 472                  $this->debug('adding array '.$this->message[$pos]['name']);
 473                  foreach($children as $child_pos){
 474                      $params[] = &$this->message[$child_pos]['result'];
 475                  }
 476              // apache Map type: java hashtable

 477              } elseif($this->message[$pos]['type'] == 'Map' && $this->message[$pos]['type_namespace'] == 'http://xml.apache.org/xml-soap'){
 478                  foreach($children as $child_pos){
 479                      $kv = explode("|",$this->message[$child_pos]['children']);
 480                         $params[$this->message[$kv[1]]['result']] = &$this->message[$kv[2]]['result'];
 481                  }
 482              // generic compound type

 483              //} elseif($this->message[$pos]['type'] == 'SOAPStruct' || $this->message[$pos]['type'] == 'struct') {

 484              } else {
 485                  // Apache Vector type: treat as an array

 486                  if ($this->message[$pos]['type'] == 'Vector' && $this->message[$pos]['type_namespace'] == 'http://xml.apache.org/xml-soap') {
 487                      $notstruct = 1;
 488                  } else {
 489                      // is array or struct?

 490                      foreach($children as $child_pos){
 491                          if(isset($keys) && isset($keys[$this->message[$child_pos]['name']])){
 492                              $notstruct = 1;
 493                              break;
 494                          }
 495                          $keys[$this->message[$child_pos]['name']] = 1;
 496                      }
 497                  }
 498                  //

 499                  foreach($children as $child_pos){
 500                      if(isset($notstruct)){
 501                          $params[] = &$this->message[$child_pos]['result'];
 502                      } else {
 503                          if (isset($params[$this->message[$child_pos]['name']])) {
 504                              // de-serialize repeated element name into an array

 505                              if (!is_array($params[$this->message[$child_pos]['name']])) {
 506                                  $params[$this->message[$child_pos]['name']] = array($params[$this->message[$child_pos]['name']]);
 507                              }
 508                              $params[$this->message[$child_pos]['name']][] = &$this->message[$child_pos]['result'];
 509                          } else {
 510                              $params[$this->message[$child_pos]['name']] = &$this->message[$child_pos]['result'];
 511                          }
 512                      }
 513                  }
 514              }
 515              return is_array($params) ? $params : array();
 516          } else {
 517              $this->debug('no children');
 518              if(strpos($this->message[$pos]['cdata'],'&')){
 519                  return  strtr($this->message[$pos]['cdata'],array_flip($this->entities));
 520              } else {
 521                  return $this->message[$pos]['cdata'];
 522              }
 523          }
 524      }
