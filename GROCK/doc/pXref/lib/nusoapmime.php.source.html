  39  /**

  40  *

  41  * soapclientmime client supporting 

  42  * <a href="http://www.w3.org/TR/SOAP-attachments">MIME attachments</a>.

  43  *

  44  * @author   Scott Nichol <snichol@sourceforge.net>

  45  * @author    Thanks to Guillaume and Henning Reich for posting great attachment code to the mail list

  46  * @version  $Id: nusoapmime.php,v 1.4 2004/04/13 12:04:27 snichol Exp $

  47  * @access   public

  48  */
  49  class soapclientmime extends soapclient {
  50      var $requestAttachments = array();
  51      var $responseAttachments;
  52      var $mimeContentType;
  53      
  54      /**

  55      * adds a MIME attachment to the current request.

  56      *

  57      * If the $data parameter contains an empty string, this method will read

  58      * the contents of the file named by the $filename parameter.

  59      *

  60      * If the $cid parameter is false, this method will generate the cid.

  61      *

  62      * @param string $data The data of the attachment

  63      * @param string $filename The filename of the attachment (default is empty string)

  64      * @param string $contenttype The MIME Content-Type of the attachment (default is application/octet-stream)

  65      * @param string $cid The content-id (cid) of the attachment (default is false)

  66      * @return string The content-id (cid) of the attachment

  67      * @access public

  68      */
  69      function addAttachment($data, $filename = '', $contenttype = 'application/octet-stream', $cid = false) {
  70          if (! $cid) {
  71              $cid = md5(uniqid(time()));
  72          }
  73  
  74          $info['data'] = $data;
  75          $info['filename'] = $filename;
  76          $info['contenttype'] = $contenttype;
  77          $info['cid'] = $cid;
  78          
  79          $this->requestAttachments[] = $info;
  80  
  81          return $cid;
  82      }
  83  
  84      /**

  85      * clears the MIME attachments for the current request.

  86      *

  87      * @access public

  88      */
  89      function clearAttachments() {
  90          $this->requestAttachments = array();
  91      }

  93      /**

  94      * gets the MIME attachments from the current response.

  95      *

  96      * Each array element in the return is an associative array with keys

  97      * data, filename, contenttype, cid.  These keys correspond to the parameters

  98      * for addAttachment.

  99      *

 100      * @return array The attachments.

 101      * @access public

 102      */
 103      function getAttachments() {
 104          return $this->responseAttachments;
 105      }

 107      /**

 108      * gets the HTTP body for the current request.

 109      *

 110      * @param string $soapmsg The SOAP payload

 111      * @return string The HTTP body, which includes the SOAP payload

 112      * @access protected

 113      */
 114      function getHTTPBody($soapmsg) {
 115          if (count($this->requestAttachments) > 0) {
 116              $params['content_type'] = 'multipart/related';
 117              $mimeMessage =& new Mail_mimePart('', $params);
 118              unset($params);
 119  
 120              $params['content_type'] = 'text/xml';
 121              $params['encoding']     = '8bit';
 122              $params['charset']      = $this->soap_defencoding;
 123              $mimeMessage->addSubpart($soapmsg, $params);
 124              
 125              foreach ($this->requestAttachments as $att) {
 126                  unset($params);
 127  
 128                  $params['content_type'] = $att['contenttype'];
 129                  $params['encoding']     = 'base64';
 130                  $params['disposition']  = 'attachment';
 131                  $params['dfilename']    = $att['filename'];
 132                  $params['cid']          = $att['cid'];
 133  
 134                  if ($att['data'] == '' && $att['filename'] <> '') {
 135                      if ($fd = fopen($att['filename'], 'rb')) {
 136                          $data = fread($fd, filesize($att['filename']));
 137                          fclose($fd);
 138                      } else {
 139                          $data = '';
 140                      }
 141                      $mimeMessage->addSubpart($data, $params);
 142                  } else {
 143                      $mimeMessage->addSubpart($att['data'], $params);
 144                  }
 145              }
 146  
 147              $output = $mimeMessage->encode();
 148              $mimeHeaders = $output['headers'];
 149      
 150              foreach ($mimeHeaders as $k => $v) {
 151                  $this->debug("MIME header $k: $v");
 152                  if (strtolower($k) == 'content-type') {
 153                      // PHP header() seems to strip leading whitespace starting

 154                      // the second line, so force everything to one line

 155                      $this->mimeContentType = str_replace("\r\n", " ", $v);
 156                  }
 157              }
 158      
 159              return $output['body'];
 160          }
 161  
 162          return parent::getHTTPBody($soapmsg);
 163      }

 165      /**

 166      * gets the HTTP content type for the current request.

 167      *

 168      * Note: getHTTPBody must be called before this.

 170      * @return string the HTTP content type for the current request.

 173      function getHTTPContentType() {
 174          if (count($this->requestAttachments) > 0) {
 175              return $this->mimeContentType;
 176          }
 177          return parent::getHTTPContentType();
 178      }

 180      /**

 181      * gets the HTTP content type charset for the current request.

 184      * Note: getHTTPBody must be called before this.

 185      *

 186      * @return string the HTTP content type charset for the current request.

 187      * @access protected

 189      function getHTTPContentTypeCharset() {
 190          if (count($this->requestAttachments) > 0) {
 191              return false;
 192          }
 193          return parent::getHTTPContentTypeCharset();
 194      }

 196      /**

 197      * processes SOAP message returned from server

 198      *

 199      * @param    array    $headers    The HTTP headers

 200      * @param    string    $data        unprocessed response data from server

 201      * @return    mixed    value of the message, decoded into a PHP type

 202      * @access   protected

 203      */
 204      function parseResponse($headers, $data) {
 205          $this->debug('Entering parseResponse() for payload of length ' . strlen($data) . ' and type of ' . $headers['content-type']);
 206          $this->responseAttachments = array();
 207          if (strstr($headers['content-type'], 'multipart/related')) {
 208              $this->debug('Decode multipart/related');
 209              $input = '';
 210              foreach ($headers as $k => $v) {
 211                  $input .= "$k: $v\r\n";
 212              }
 213              $params['input'] = $input . "\r\n" . $data;
 214              $params['include_bodies'] = true;
 215              $params['decode_bodies'] = true;
 216              $params['decode_headers'] = true;
 217              
 218              $structure = Mail_mimeDecode::decode($params);
 219  
 220              foreach ($structure->parts as $part) {
 221                  if (!isset($part->disposition)) {
 222                      $this->debug('Have root part of type ' . $part->headers['content-type']);
 223                      $return = parent::parseResponse($part->headers, $part->body);
 224                  } else {
 225                      $this->debug('Have an attachment of type ' . $part->headers['content-type']);
 226                      $info['data'] = $part->body;
 227                      $info['filename'] = isset($part->d_parameters['filename']) ? $part->d_parameters['filename'] : '';
 228                      $info['contenttype'] = $part->headers['content-type'];
 229                      $info['cid'] = $part->headers['content-id'];
 230                      $this->responseAttachments[] = $info;
 231                  }
 232              }
 233          
 234              if (isset($return)) {
 235                  return $return;
 236              }
 237              
 238              $this->setError('No root part found in multipart/related content');
 239              return;
 240          }
 241          $this->debug('Not multipart/related');
 242          return parent::parseResponse($headers, $data);
 243      }
