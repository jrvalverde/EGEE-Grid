 194      /**

 195      * send the SOAP message via HTTP

 196      *

 197      * @param    string $data message data

 198      * @param    integer $timeout set connection timeout in seconds

 199      * @param    integer $response_timeout set response timeout in seconds

 200      * @return    string data

 201      * @access   public

 202      */
 203      function send($data, $timeout=0, $response_timeout=30) {
 204          
 205          $this->debug('entered send() with data of length: '.strlen($data));
 206  
 207          $this->tryagain = true;
 208          $tries = 0;
 209          while ($this->tryagain) {
 210              $this->tryagain = false;
 211              if ($tries++ < 2) {
 212                  // make connnection

 213                  if (!$this->connect($timeout, $response_timeout)){
 214                      return false;
 215                  }
 216                  
 217                  // send request

 218                  if (!$this->sendRequest($data)){
 219                      return false;
 220                  }
 221                  
 222                  // get response

 223                  $respdata = $this->getResponse();
 224              } else {
 225                  $this->setError('Too many tries to get an OK response');
 226              }
 227          }        
 228          $this->debug('end of send()');
 229          return $respdata;
 230      }

 255      function setCredentials($username, $password, $authtype = 'basic', $digestRequest = array()) {
 256          global $_SERVER;
 257  
 258          $this->debug("Set credentials for authtype $authtype");
 259          // cf. RFC 2617

 260          if ($authtype == 'basic') {
 261              $this->outgoing_headers['Authorization'] = 'Basic '.base64_encode($username.':'.$password);
 262          } elseif ($authtype == 'digest') {
 263              if (isset($digestRequest['nonce'])) {
 264                  $digestRequest['nc'] = isset($digestRequest['nc']) ? $digestRequest['nc']++ : 1;
 265                  
 266                  // calculate the Digest hashes (calculate code based on digest implementation found at: http://www.rassoc.com/gregr/weblog/stories/2002/07/09/webServicesSecurityHttpDigestAuthenticationWithoutActiveDirectory.html)

 267      
 268                  // A1 = unq(username-value) ":" unq(realm-value) ":" passwd

 269                  $A1 = $username. ':' . $digestRequest['realm'] . ':' . $password;
 270      
 271                  // H(A1) = MD5(A1)

 272                  $HA1 = md5($A1);
 273      
 274                  // A2 = Method ":" digest-uri-value

 275                  $A2 = 'POST:' . $this->uri;
 276      
 277                  // H(A2)

 278                  $HA2 =  md5($A2);
 279      
 280                  // KD(secret, data) = H(concat(secret, ":", data))

 281                  // if qop == auth:

 282                  // request-digest  = <"> < KD ( H(A1),     unq(nonce-value)

 283                  //                              ":" nc-value

 284                  //                              ":" unq(cnonce-value)

 285                  //                              ":" unq(qop-value)

 286                  //                              ":" H(A2)

 287                  //                            ) <">

 288                  // if qop is missing,

 289                  // request-digest  = <"> < KD ( H(A1), unq(nonce-value) ":" H(A2) ) > <">

 290      
 291                  $unhashedDigest = '';
 292                  $nonce = isset($digestRequest['nonce']) ? $digestRequest['nonce'] : '';
 293                  $cnonce = $nonce;
 294                  if ($digestRequest['qop'] != '') {
 295                      $unhashedDigest = $HA1 . ':' . $nonce . ':' . sprintf("%08d", $digestRequest['nc']) . ':' . $cnonce . ':' . $digestRequest['qop'] . ':' . $HA2;
 296                  } else {
 297                      $unhashedDigest = $HA1 . ':' . $nonce . ':' . $HA2;
 298                  }
 299      
 300                  $hashedDigest = md5($unhashedDigest);
 301      
 302                  $this->outgoing_headers['Authorization'] = 'Digest username="' . $username . '", realm="' . $digestRequest['realm'] . '", nonce="' . $nonce . '", uri="' . $this->uri . '", cnonce="' . $cnonce . '", nc=' . sprintf("%08x", $digestRequest['nc']) . ', qop="' . $digestRequest['qop'] . '", response="' . $hashedDigest . '"';
 303              }
 304          }
 305          $this->username = $username;
 306          $this->password = $password;
 307          $this->authtype = $authtype;
 308          $this->digestRequest = $digestRequest;
 309          
 310          if (isset($this->outgoing_headers['Authorization'])) {
 311              $this->debug('Authorization header set: ' . substr($this->outgoing_headers['Authorization'], 0, 12) . '...');
 312          } else {
 313              $this->debug('Authorization header not set');
 314          }
 315      }

 361      /**

 362      * decode a string that is encoded w/ "chunked' transfer encoding

 363       * as defined in RFC2068 19.4.6

 364      *

 365      * @param    string $buffer

 366      * @param    string $lb

 367      * @returns    string

 368      * @access   public

 369      */
 370      function decodeChunked($buffer, $lb){
 371          // length := 0

 372          $length = 0;
 373          $new = '';
 374          
 375          // read chunk-size, chunk-extension (if any) and CRLF

 376          // get the position of the linebreak

 377          $chunkend = strpos($buffer, $lb);
 378          if ($chunkend == FALSE) {
 379              $this->debug('no linebreak found in decodeChunked');
 380              return $new;
 381          }
 382          $temp = substr($buffer,0,$chunkend);
 383          $chunk_size = hexdec( trim($temp) );
 384          $chunkstart = $chunkend + strlen($lb);
 385          // while (chunk-size > 0) {

 386          while ($chunk_size > 0) {
 387              $this->debug("chunkstart: $chunkstart chunk_size: $chunk_size");
 388              $chunkend = strpos( $buffer, $lb, $chunkstart + $chunk_size);
 389                
 390              // Just in case we got a broken connection

 391                if ($chunkend == FALSE) {
 392                    $chunk = substr($buffer,$chunkstart);
 393                  // append chunk-data to entity-body

 394                  $new .= $chunk;
 395                    $length += strlen($chunk);
 396                    break;
 397              }
 398              
 399                // read chunk-data and CRLF

 400                $chunk = substr($buffer,$chunkstart,$chunkend-$chunkstart);
 401                // append chunk-data to entity-body

 402                $new .= $chunk;
 403                // length := length + chunk-size

 404                $length += strlen($chunk);
 405                // read chunk-size and CRLF

 406                $chunkstart = $chunkend + strlen($lb);
 407              
 408                $chunkend = strpos($buffer, $lb, $chunkstart) + strlen($lb);
 409              if ($chunkend == FALSE) {
 410                  break; //Just in case we got a broken connection
 411              }
 412              $temp = substr($buffer,$chunkstart,$chunkend-$chunkstart);
 413              $chunk_size = hexdec( trim($temp) );
 414              $chunkstart = $chunkend;
 415          }
 416          return $new;
 417      }

 474      function getResponse(){
 475          $this->incoming_payload = '';
 476          
 477        if ($this->scheme == 'http' || $this->scheme == 'ssl') {
 478          // loop until headers have been retrieved

 479          $data = '';
 480          while (!isset($lb)){
 481  
 482              // We might EOF during header read.

 483              if(feof($this->fp)) {
 484                  $this->incoming_payload = $data;
 485                  $this->debug('found no headers before EOF after length ' . strlen($data));
 486                  $this->debug("received before EOF:\n" . $data);
 487                  $this->setError('server failed to send headers');
 488                  return false;
 489              }
 490  
 491              $tmp = fgets($this->fp, 256);
 492              $tmplen = strlen($tmp);
 493              $this->debug("read line of $tmplen bytes: " . trim($tmp));
 494  
 495              if ($tmplen == 0) {
 496                  $this->incoming_payload = $data;
 497                  $this->debug('socket read of headers timed out after length ' . strlen($data));
 498                  $this->debug("read before timeout:\n" . $data);
 499                  $this->setError('socket read of headers timed out');
 500                  return false;
 501              }
 502  
 503              $data .= $tmp;
 504              $pos = strpos($data,"\r\n\r\n");
 505              if($pos > 1){
 506                  $lb = "\r\n";
 507              } else {
 508                  $pos = strpos($data,"\n\n");
 509                  if($pos > 1){
 510                      $lb = "\n";
 511                  }
 512              }
 513              // remove 100 header

 514              if(isset($lb) && ereg('^HTTP/1.1 100',$data)){
 515                  unset($lb);
 516                  $data = '';
 517              }//
 518          }
 519          // store header data

 520          $this->incoming_payload .= $data;
 521          $this->debug('found end of headers after length ' . strlen($data));
 522          // process headers

 523          $header_data = trim(substr($data,0,$pos));
 524          $header_array = explode($lb,$header_data);
 525          $this->incoming_headers = array();
 526          foreach($header_array as $header_line){
 527              $arr = explode(':',$header_line, 2);
 528              if(count($arr) > 1){
 529                  $header_name = strtolower(trim($arr[0]));
 530                  $this->incoming_headers[$header_name] = trim($arr[1]);
 531              } else if (isset($header_name)) {
 532                  $this->incoming_headers[$header_name] .= $lb . ' ' . $header_line;
 533              }
 534          }
 535          
 536          // loop until msg has been received

 537          if (isset($this->incoming_headers['content-length'])) {
 538              $content_length = $this->incoming_headers['content-length'];
 539              $chunked = false;
 540              $this->debug("want to read content of length $content_length");
 541          } else {
 542              $content_length =  2147483647;
 543              if (isset($this->incoming_headers['transfer-encoding']) && strtolower($this->incoming_headers['transfer-encoding']) == 'chunked') {
 544                  $chunked = true;
 545                  $this->debug("want to read chunked content");
 546              } else {
 547                  $chunked = false;
 548                  $this->debug("want to read content to EOF");
 549              }
 550          }
 551          $data = '';
 552          do {
 553              if ($chunked) {
 554                  $tmp = fgets($this->fp, 256);
 555                  $tmplen = strlen($tmp);
 556                  $this->debug("read chunk line of $tmplen bytes");
 557                  if ($tmplen == 0) {
 558                      $this->incoming_payload = $data;
 559                      $this->debug('socket read of chunk length timed out after length ' . strlen($data));
 560                      $this->debug("read before timeout:\n" . $data);
 561                      $this->setError('socket read of chunk length timed out');
 562                      return false;
 563                  }
 564                  $content_length = hexdec(trim($tmp));
 565                  $this->debug("chunk length $content_length");
 566              }
 567              $strlen = 0;
 568              while (($strlen < $content_length) && (!feof($this->fp))) {
 569                  $readlen = min(8192, $content_length - $strlen);
 570                  $tmp = fread($this->fp, $readlen);
 571                  $tmplen = strlen($tmp);
 572                  $this->debug("read buffer of $tmplen bytes");
 573                  if (($tmplen == 0) && (!feof($this->fp))) {
 574                      $this->incoming_payload = $data;
 575                      $this->debug('socket read of body timed out after length ' . strlen($data));
 576                      $this->debug("read before timeout:\n" . $data);
 577                      $this->setError('socket read of body timed out');
 578                      return false;
 579                  }
 580                  $strlen += $tmplen;
 581                  $data .= $tmp;
 582              }
 583              if ($chunked && ($content_length > 0)) {
 584                  $tmp = fgets($this->fp, 256);
 585                  $tmplen = strlen($tmp);
 586                  $this->debug("read chunk terminator of $tmplen bytes");
 587                  if ($tmplen == 0) {
 588                      $this->incoming_payload = $data;
 589                      $this->debug('socket read of chunk terminator timed out after length ' . strlen($data));
 590                      $this->debug("read before timeout:\n" . $data);
 591                      $this->setError('socket read of chunk terminator timed out');
 592                      return false;
 593                  }
 594              }
 595          } while ($chunked && ($content_length > 0) && (!feof($this->fp)));
 596          if (feof($this->fp)) {
 597              $this->debug('read to EOF');
 598          }
 599          $this->debug('read body of length ' . strlen($data));
 600          $this->incoming_payload .= $data;
 601          $this->debug('received a total of '.strlen($this->incoming_payload).' bytes of data from server');
 602          
 603          // close filepointer

 604          if(
 605              (isset($this->incoming_headers['connection']) && strtolower($this->incoming_headers['connection']) == 'close') || 
 606              (! $this->persistentConnection) || feof($this->fp)){
 607              fclose($this->fp);
 608              $this->fp = false;
 609              $this->debug('closed socket');
 610          }
 611
