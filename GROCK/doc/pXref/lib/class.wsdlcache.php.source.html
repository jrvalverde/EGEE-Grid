   6  * caches instances of the wsdl class

   7  * 

   8  * @author   Scott Nichol <snichol@computer.org>

   9  * @author    Ingo Fischer <ingo@apollon.de>

  13  class wsdlcache {
  14      var $fplock;
  15      var $cache_lifetime;
  16      var $cache_dir;
  17      var $debug_str = '';
  18  
  19      /*

  20      * constructor

  21      *

  22      * @param string $cache_dir directory for cache-files

  23      * @param integer $cache_lifetime lifetime for caching-files in seconds or 0 for unlimited

  24      * @access public

  25      */
  26      function wsdlcache($cache_dir='.', $cache_lifetime=0) {
  27          $this->fplock = array();
  28          $this->cache_dir = $cache_dir != '' ? $cache_dir : '.';
  29          $this->cache_lifetime = $cache_lifetime;
  30      }
  31  
  32      /**

  33      * creates the filename used to cache a wsdl instance

  34      *

  35      * @param string $wsdl The URL of the wsdl instance

  36      * @return string The filename used to cache the instance

  37      * @access protected

  38      */
  39      function createFilename($wsdl) {
  40          return $this->cache_dir.'/wsdlcache-' . md5($wsdl);
  41      }
  42  
  43      /**

  44      * adds debug data to the class level debug string

  45      *

  46      * @param    string $string debug data

  47      * @access   private

  48      */
  49      function debug($string){
  50          $this->debug_str .= get_class($this).": $string\n";
  51      }
  52  
  53      /**

  54      * gets a wsdl instance from the cache

  55      *

  56      * @param string $wsdl The URL of the wsdl instance

  57      * @return object The cached wsdl instance, null if the instance is not in the cache

  58      * @access public

  59      */
  60      function get($wsdl) {
  61          $filename = $this->createFilename($wsdl);
  62          if ($this->obtainMutex($filename, "r")) {
  63              // check for expired WSDL that must be removed from the cache

  64               if ($this->cache_lifetime > 0) {
  65                  if (file_exists($filename) && (time() - filemtime($filename) > $this->cache_lifetime)) {
  66                      unlink($filename);
  67                      $this->debug("Expired $wsdl ($filename) from cache");
  68                      $this->releaseMutex($filename);
  69                      return null;
  70                    }
  71              }
  72              // see what there is to return

  73              $fp = @fopen($filename, "r");
  74              if ($fp) {
  75                  $s = implode("", @file($filename));
  76                  fclose($fp);
  77                  $this->debug("Got $wsdl ($filename) from cache");
  78              } else {
  79                  $s = null;
  80                  $this->debug("$wsdl ($filename) not in cache");
  81              }
  82              $this->releaseMutex($filename);
  83              return (!is_null($s)) ? unserialize($s) : null;
  84          }
  85          return null;
  86      }
  87  
  88      /**

  89      * obtains the local mutex

  90      *

  91      * @param string $filename The Filename of the Cache to lock

  92      * @param string $mode The open-mode ("r" or "w") or the file - affects lock-mode

  93      * @return boolean Lock successfully obtained ?!

  94      * @access protected

  95      */
  96      function obtainMutex($filename, $mode) {
  97          if (isset($this->fplock[md5($filename)])) {
  98              $this->debug("Lock for $filename already exists");
  99              return false;
 100          }
 101          $this->fplock[md5($filename)] = fopen($filename.".lock", "w");
 102          if ($mode == "r") {
 103              return flock($this->fplock[md5($filename)], LOCK_SH);
 104          } else {
 105              return flock($this->fplock[md5($filename)], LOCK_EX);
 106          }
 107      }

 109      /*

 110      * adds a wsdl instance to the cache

 111      *

 112      * @param object $wsdl_instance The wsdl instance to add

 113      * @return boolean WSDL successfully cached

 114      * @access public

 115      */
 116      function put($wsdl_instance) {
 117          $filename = $this->createFilename($wsdl_instance->wsdl);
 118          $s = serialize($wsdl_instance);
 119          if ($this->obtainMutex($filename, "w")) {
 120              $fp = fopen($filename, "w");
 121              fputs($fp, $s);
 122              fclose($fp);
 123              $this->debug("Put $wsdl_instance->wsdl ($filename) in cache");
 124              $this->releaseMutex($filename);
 125              return true;
 126          }
 127          return false;
 128      }
 129  
 130      /**

 131      * releases the local mutex

 132      *

 133      * @param string $filename The Filename of the Cache to lock

 134      * @return boolean Lock successfully released

 135      * @access protected

 136      */
 137      function releaseMutex($filename) {
 138          $ret = flock($this->fplock[md5($filename)], LOCK_UN);
 139          fclose($this->fplock[md5($filename)]);
 140          unset($this->fplock[md5($filename)]);
 141          return $ret;
 142      }
 143  
 144      /**

 145      * removes a wsdl instance from the cache

 146      *

 147      * @param string $wsdl The URL of the wsdl instance

 148      * @return boolean Whether there was an instance to remove

 149      * @access public

 150      */
 151      function remove($wsdl) {
 152          $filename = $this->createFilename($wsdl);
 153          $this->obtainMutex($filename, "w");
 154          $ret = unlink($filename);
 155          $this->debug("Removed $wsdl ($filename) from cache");
 156          $this->releaseMutex($filename);
 157          return $ret;
 158      }
