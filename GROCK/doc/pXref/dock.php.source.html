  45  /**
  46   *  Create a docking job in the current directory
  47   *
  48   *  Jobs are created as subdirectories inside the current directory so
  49   * that each of them is independent and self-contained.
  50   *
  51   *  Since we support various docking methods, we just switch over all of
  52   * them.
  53   *
  54   *  Since we don't want to concern ourselves with the diferent kinds of
  55   * options supported by each method, we just pass them along to the
  56   * underlying method and leave to it the parameter verification.
  57   *
  58   *  We'll return whatever the underlying method does.
  59   *
  60   *  NOTE: these functions expect actual data to be located
  61   *  always in $session_id/ligand.pdb and $session_id/receptor.pdb
  62   *  independent of the names they receive.
  63   *
  64   *      $session_id/ligand.pdb should be the smaller of the two
  65   *          molecules, which is to be moved around the bigger
  66   *      $session_id/receptor.pdb should be the bigger, which is
  67   *          to remain statically fixed during the docking process.
  68   *
  69   *  There is NO WAY to associate $user_probe or $db_target with
  70   *  the actual filenames (ligand.pdb or receptor.pdb) in the
  71   *  current implementation.
  72   *
  73   *  @param  $probe  the user-specified probe molecule name
  74   *  @param  $target the target molecule name
  75   *  @param  $options    the options selected for docking (method and parameters)
  76   *
  77   *  @return  exit status of low-level job creation function.
  78   */
  79  function dock_create_job($user_probe, $db_target, $options)
  80  {
  81      // we just switch over available methods, defaulting to GRAMM
  82      if (strcmp($options['docker'], 'gramm') == 0) {
  83      return gramm_create_job($user_probe, $db_target, $options);
  84      } else {
  85          /* default case */
  86      error("create dock job", "unknown method ".$options['docker']);
  87      return FALSE;
  88      }
  89  }

  91  /**
  92   *  Generate scores files
  93   *
  94   *  This routine evaluates the output got to date and generates one or
  95   * more scores files:
  96   *
  97   * - scores.txt containing all the sorted scores for all matches between the
  98   *  probe molecule and all target molecules screened from the database
  99   * - top_scores.txt containing only the topmost score for each match of
 100   *  the probe molecule against each molecules screened from the database
 101   *  sorted by energy.
 102   *
 103   *  As the scores files and their contents and format will be dependent
 104   * on the docking method used, we just pass on the task to the appropriate
 105   * routine for the method specified.
 106   *
 107   *  @param string $docker   The docking method selected by the user
 108   */
 109  function dock_generate_scores($docker)
 110  {
 111      switch ($docker) {
 112          case 'gramm':
 113      case '3ddock':
 114      case 'dock5':
 115      case 'dot':
 116              $func = $docker."_generate_scores";
 117              $func($docker);
 118          break;
 119      default:
 120          error("Generate scores", "Unknown method");
 121          return;
 122      }
 123  }

 125  /**
 126   *  Show a selection from the listing of top scores
 127   *
 128   *  Since scores are dependent on the method, we relay to the docker the task
 129   * of knowing how to deal with score presentation to the user.
 130   *
 131   *  @param string $docker   The program used to perform the match (used to process
 132   *                          job output)
 133   *  @param string $probe    The name of the probe molecule submitted by the user
 134   *  @param string $pt        How to treat the probe molecule (ligand or receptor)
 135   *  @param integer $from    Top result at which display should start
 136   *  @param integer $to      Top result at which display should end
 137   */
 138  function dock_show_top_scores($docker, $id, $probe, $pt, $from=1, $to=50)
 139  {
 140      if (($from <= 0) || ($to <= 0)) {
 141          error("Show top scores", "I can't show non-existing matches $from .. $to");
 142      }
 143          
 144      if ($from > $to) {
 145          // swap
 146      $aux = $to;
 147      $to = $from;
 148      $from = $aux;
 149      }
 150  
 151      // XXX JR XXX Caution, while this may work it doesn't check for
 152      // validity! This is just a proof of concept! ;-)
 153      switch ($docker) {
 154          case 'gramm':
 155      case '3ddock':
 156      case 'dock5':
 157      case 'dot':
 158              $func = $docker."_show_top_scores";
 159              $func($docker, $id, $probe, $pt, $from, $to);
 160          break;
 161      default:
 162          error("Show top scores", "Unknown method");
 163          return;
 164      }
 165  }

 191      if (($from <= 0) || ($to <= 0)) {
 192          error("Show dock", "I can't show non-existing matches $from .. $to");
 193      }
 194          
 195      if ($from > $to) {
 196          // swap
 197      $aux=$to;
 198      $to=$from;
 199      $from=$aux;
 200      }
 201      
 202      switch ($docker) {
 203          case 'gramm':
 204      case '3ddock':
 205      case 'dock5':
 206      case 'dot':
 207          $func = $docker."_show_dock";
 208          $func($docker, $id, $probe, $pt, $target, $from, $to);
 209          break;
 210      default:
 211          error("Show docking of $probe against $target", "Unknown method");
 212          return;
 213      }
