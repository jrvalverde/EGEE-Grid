  39  /**
  40   *  Create a GRAMM job
  41   *
  42   *  This functions sets up a GRAMM job for submission to the Grid.
  43   * The job created will match a mobile molecule against a fixed
  44   * one (where the mobile should be the smaller one and the fixed
  45   * the bigger for efficiency).
  46   *
  47   *  In order to submit a job, it must be self-contained: we create
  48   * a directory named after the probe and target molecules and store
  49   * there all info needed:
  50   *      - the GRAMM command
  51   *      - GRAMM configuration files (according to user preferences)
  52   *      - the ligand/mobile molecule
  53   *      - the receptor/fixed molecule
  54   *      - a shell script to drive the execution
  55   *      - the JDL describing the job
  56   *
  57   *  We'll take the executable and most of the configuration files
  58   * from the application directory (where we are installed, $app_dir).
  59   *
  60   *  For simplicity, we will use the molecules called 'ligand.pdb'
  61   * (the mobile, smaller one) and 'receptor.pdb' (the bigger one),
  62   * and we expect them to be available in the current directory.
  63   *
  64   *  We might as well use "$probe.pdb" and "$target.pdb" as the file
  65   * names, but this has two implications, a minor and a major one:
  66   *
  67   *  The minor one is that we would have to tweak other config files
  68   * as well and use $probe+$target as job name.
  69   *
  70   *  The major one is that we would need to pass this information to
  71   * the show_results.php script, which is run separately and independently
  72   * from us, making the logic somewhat more convoluted (for the time
  73   * being; may be we could pass this as hidden form variables?).
  74   *
  75   *  NOTE: these functions expect actual data to be located
  76   *  always in ./ligand.pdb and ./receptor.pdb
  77   *  independent of the names they receive.
  78   *
  79   *      ./ligand.pdb should be the smaller of the two
  80   *          molecules, which is to be moved around the bigger
  81   *      ./receptor.pdb should be the bigger, which is
  82   *          to remain statically fixed during the docking process.
  83   *
  84   *  There is NO WAY to associate $user_probe or $db_target with
  85   *  the actual filenames (ligand.pdb or receptor.pdb) in the
  86   *  current implementation.
  87   *
  88   *  @param $probe  is the original user-provided probe molecule name
  89   *  @param $target   is the original name of the db molecule being
  90   *                  screened
  91   *  @param $options The user selected options
  92   *
  93   *  @return boolean TRUE is all went well, FALSE otherwise.
  94   *
  95   */
  96  function gramm_create_job($probe, $target, $options)
  97  {
  98      global $debug;
  99      global $app_dir;        // where exec and config files are located
 100      global $output;
 101  
 102      if (strcmp($options['docker'], 'gramm') != 0) {
 103          echo_error( "create gramm job", "method selected is not GRAMM");
 104          return FALSE;
 105      }
 106      
 107      if ($debug) fwrite($output, "<h3>Creating a GRAMM job for $probe+$target</h3>\n");
 108      // To launch a job to the grid it must be fully self-contained
 109      // within its own subdirectory.
 110      // Create job directory
 111      // We use $target as the name to identify the job.
 112      $job_directory='./'.$target;
 113      if (! mkdir($job_directory, 0700)) {
 114          echo_error("create gramm job", 
 115          "could not create directory for $probe+$target");
 116      return FALSE;
 117      }
 118          
 119      // Copy the ligand file from the current directory
 120      // into the job directory
 121      if (!copy("ligand.pdb", "$job_directory/ligand.pdb")) {
 122          echo_error("create gramm job", 
 123          "could not make a copy of ligand to dock $probe+$target");
 124          return FALSE;
 125      }
 126      
 127      // Copy the receptor file from the session directory
 128      // into the job directory
 129      if (!copy("receptor.pdb", "$job_directory/receptor.pdb")) {
 130          echo_error("create gramm job", 
 131          "could not make a copy of ligand to dock $probe+$target");
 132          return FALSE;
 133      }
 134       
 135      // Copy GRAMM program and configuration files. 
 136      // $app_dir/gramm is a reference template job directory containing
 137      //    the program, needed aux files, config files and JDL files.
 138      exec("cp $app_dir/gramm/*.* $job_directory/.", $out, $exit);
 139      if ($exit != 0) {
 140          echo_error("create gramm job", 
 141          "couldn't install job files for $target");
 142      fwrite($output, print_r($out, TRUE));
 143      return FALSE;
 144      }
 145      // rpar.gr is generated under the user instructions from the submit form.
 146      if (!write_rpar_gr($job_directory, $options)) {
 147          echo_error("create_gramm job",
 148          "couldn't configure job $target");
 149      return FALSE;
 150      }
 151      // copy JDL file describing the job to the grid
 152      // XXX JR XXX no longer needed, moved into gramm directory
 153      //if (! copy("$app_dir/jdl/session.jdl", "$job_directory/session.jdl")) {
 154      //    echo_error("Create GRAMM job",
 155      //        "couldn't install JDL file for $probe+$target");
 156      //    return FALSE;
 157      //}
 158  
 159      // Generate the gramm-go.tar.gz package
 160      chdir($job_directory);
 161      exec("tar -cf gramm-go.tar *.dat *.gr gramm.lnx", $out, $exit);
 162      if ($exit != 0) {
 163          echo_error("Create GRAMM job",
 164          "couldn't create job package for $probe+$target");
 165      fwrite($output, print_r($out, TRUE));
 166      chdir('..');
 167      return FALSE;
 168      }
 169      unset($out);
 170      exec("gzip gramm-go.tar", $out, $exit);
 171      if ($exit == 1) {
 172          echo_error("Create GRAMM job",
 173          "couldn't compress the job package $probe+$target ($exit)");
 174      fwrite($output, print_r($out, TRUE));
 175      chdir('..');
 176      return FALSE;
 177      }
 178      unset($out);
 179      // clean up
 180      // exec("rm -f *dat *gr gramm", $out, $exit);
 181      // for some reason, gzip is leaving the tar file behind:
 182      exec("rm -f *dat *gr gramm.lnx gramm-go.tar", $out, $exit);
 183      if ($exit != 0) {
 184          echo_warning("Create GRAMM job",
 185          "couldn't remove unneeded files from $probe+$target");
 186      fwrite($output, print_r($out, TRUE));
 187      }
 188      unset($out);
 189      chdir("..");
 190      
 191      return TRUE;
 192  }

 194  /**
 195   *  Write parameter file to drive the docking job.
 196   *
 197   *  NOTE currently there is no error checking. If an fwrite fails we won't
 198   * notice.
 199   *
 200   *  @access private
 201   *
 202   *  @param  $directory where we want it located
 203   *  @param  $options array containing user selected preferences
 204   *
 205   *  @return boolean TRUE if all went well, FALSE otherwise
 206   */
 207  function write_rpar_gr($directory, $options)
 208  {
 209      global $debug;
 210  
 211      $rpar="$directory/rpar.gr";
 212      $matching = "generic"; // or $matching = "helix";
 213      $resolution = $options['resolution'];
 214      $representation = $options['representation'];
 215      
 216      if ($representation=="hydrophobic")
 217      {
 218          $projection="blackwhite";
 219      }else
 220      {
 221          $projection="gray";
 222      }
 223  
 224      $fp = fopen( "$rpar", "w" );
 225  
 226      if ( $resolution=="low") {
 227          $gstep = "6.8";
 228      $repulsion = "6.5";
 229      $crang = "grid_step";
 230      } else {
 231          $gstep = "1.7";
 232      $repulsion = "30.";
 233      $crang = "atom_radius";
 234      }
 235      fwrite($fp, "Matching mode (generic/helix) ....................... mmode= $matching\n" );
 236      fwrite($fp, "Grid step ............................................. eta= $gstep\n" );
 237      fwrite($fp, "Repulsion (attraction is always -1) .................... ro= $repulsion\n" );
 238      fwrite($fp, "Attraction double range (fraction of single range) ..... fr= 0.\n" );
 239      fwrite($fp, "Potential range type (atom_radius, grid_step) ....... crang= $crang\n" );
 240      fwrite($fp, "Projection (blackwhite, gray) ................ ....... ccti= $projection\n" );
 241      fwrite($fp, "Representation (all, hydrophobic) .................... crep= $representation\n" );
 242      fwrite($fp, "Number of matches to output .......................... maxm= 1000\n" );
 243      fwrite($fp, "Angle for rotations, deg (10,12,15,18,20,30, 0-no rot.)  ai= 20\n" );
 244  
 245      fclose($fp);
 246      // tired of error checking... wish we could catch exceptions!
 247      return TRUE;
 248  }

 250  /**
 251   *  Generate scores files
 252   *
 253   *  This routine evaluates the output got to date and generates one or
 254   * more scores files:
 255   *
 256   * - scores.txt containing all the sorted scores for all matches between the
 257   *  probe molecule and all target molecules screened from the database
 258   * - top_scores.txt containing only the topmost score for each match of
 259   *  the probe molecule against each molecules screened from the database
 260   *  sorted by energy.
 261   *
 262   *  Note that it may silently fail to generate any scores at all, e.g.
 263   * if no results are yet available, but also if there is an error.
 264   *
 265   *  @param string $docker   The docking method selected by the user
 266   */
 267  function gramm_generate_scores($docker)
 268  {
 269      global $output;
 270      global $app_dir;
 271      global $debug;
 272      
 273      if (strcmp($docker, "gramm") != 0) {
 274          echo_error("GRAMM - generate scores",
 275          "cannot process non-GRAMM results");
 276      return;
 277      }
 278  
 279      // generate scores file including the target entry description
 280      $target_list = fopen('target_list.txt', 'r');
 281      if ($target_list == FALSE) return;
 282      
 283      while (!feof($target_list)) {
 284          $line = fgets($target_list);
 285          if (trim($line, "\n") == '') continue;
 286      $target = strtok($line, ' ');    // ENTR_C (entry_chain)
 287          $description = strtok("\n");       // description
 288      if (file_exists("$target/job_output/gramm-come.tar.gz")) {
 289          exec("tar -zxOf $target/job_output/gramm-come.tar.gz ".
 290               "     receptor-ligand.res | ".
 291                   "tail -n +32 | ".
 292                   "sed -e 's#\$#  $target $description#g' >> sctmp", $out, $ret);  
 293      }
 294      exec("echo \"\" >> sctmp",$out, $ret);
 295      }
 296      exec('sort -r -n -k 2 sctmp > scores.txt', $out, $ret);
 297      unlink('sctmp');
 298      
 299  
 300      // generate top_scores file
 301      if ($debug) fwrite($output,
 302      "grep \"^   1\" scores.txt > top_scores.txt\n");
 303      exec("grep \"^   1\" scores.txt > top_scores.txt 2> /dev/null", $out, $ret);
 304      
 305  }

 320  function gramm_show_top_scores($docker, $id, $probe, $pt, $from=1, $to=GRAMM_DISPLAY)
 321  {
 322      global $debug;
 323      global $www_tmp;
 324      global $www_app_dir;
 325      global $app_dir;
 326      global $pdb2vrml1;
 327      global $pdb2vrml2;

 335      return;
 336      }
 337      
 338      echo "<center><table border=\"1\" cellspacing=\"0\" width =\"90%\" ".
 339       "style=\"background-image:url('$www_app_dir/images/none.png'); ".
 340               "background-color:white; border:2 ridge #800000\"".
 341           ">\n";
 342      echo "<tr>".
 343           "  <th bgcolor=\"cyan\">Match</th>\n".
 344           "  <th bgcolor=\"pink\">Energy (-)</th>\n".
 345       "  <th bgcolor=\"white\">Description</th>\n".
 346       "  <th bgcolor=\"white\">Explore</th>\n".
 347       "</tr>\n";
 348  
 349      //$reldir = "$www_tmp/$id/$target/job_output/";
 350      
 351      $top_scores = fopen("top_scores.txt", "r");
 352      if ($top_scores == FALSE) {
 353          echo_error("GRAMM - show top scores",
 354          "cannot find any scores");
 355      return;
 356      }
 357      // skip initial $from lines
 358      for ($i = 1; $i < $from; $i++) {
 359          fgets($top_scores);
 360      if (feof($top_scores)) {
 361          error("GRAMM show top scores", "Specified start position is too high");
 362          // what  to do now? END?
 363          // think this
 364      }
 365      }
 366      // show until $to line
 367      for ($i = $from; $i <= $to; $i++) {
 368          // get entry
 369          $line = fgets($top_scores);
 370      if (trim($line) == '') continue;
 371      
 372      // parse entry
 373          sscanf($line, "%d %d %d %d %d %f %f %f %s",
 374          $matchno, $energy, $rotx, $roty, $rotz, $trx, $try, $trz, $target);
 375          $target_desc = strstr($line, $target);
 376      
 377      // show data
 378      // 1. Match ligand vs. receptor
 379      echo "<tr>\n".
 380           "  <td bgcolor=\"LightGreen\">".
 381               "<strong>$probe</strong> vs. <strong>$target</strong></td>\n";
 382      // 2. Energy
 383      echo "  <td bgcolor=\"Yellow\">$energy</td>\n";
 384      // 3. Description
 385          echo "  <td>$target_desc</td>\n";
 386      // 4. Explore best 10 matches with this target
 387      echo "  <td><a href=\"$www_app_dir/script/explore_match.php?id=".
 388           basename(getcwd()).
 389           "&probe=$probe&pt=$pt&target=$target&docker=$docker\">Best 10 scores</a></td>".
 390           "</tr>\n";
 391      if (feof($top_scores))
 392          // we are done, even if we didn't reach $to
 393          break;
 394      }
 395      fclose($top_scores);
 396      echo "</table></center>\n";
 397  }
