 307  function grock_it($auth_token, $grock_options)
 308  {
 309      $auth = array();
 310      $auth['server'] = $auth_token['user'].'@'.$auth_token['host'];         // user@back-end
 311      $auth['user'] = $auth_token['user'];             // user
 312      $auth['host'] = $auth_token['host'];             // back-end
 313      $auth['password'] = $auth_token['password'];     // user@back-end passwd
 314      $auth['passphrase'] = $auth_token['passphrase']; // user grid passphrase
 315  
 316      $options = array();
 317      $options['docker'] = $grock_options['docker'];
 318      $options['database'] = $grock_options['database'];
 319      $options['probe_type'] = $grock_options['probe_type'];
 320      $options['resolution'] = $grock_options['resolution'];
 321      $options['representation'] = $grock_options['representation'];
 322  
 323      // generate session and go to it
 324      $session_id =activate_new_sandbox();
 325      
 326      // save probe as a file
 327      file_put_contents('probe.pdb', $grock_options['probe3D']);
 328  
 329      ob_end_flush();
 330      grock_set_status('starting');
 331      flush();
 332      
 333      umask(077);
 334      $grockopt = fopen("options", "w+");
 335      foreach ($auth as $idx => $value) fwrite($grockopt, "$idx=$value\n");
 336      fwrite($grockopt, "\n");
 337      foreach ($options as $idx => $value) fwrite($grockopt, "$idx=$value\n");
 338      fclose($grockopt);
 339      exec("(/usr/bin/php $app_dir/script/grock.php > grock_error 2>&1 )&");
 340  
 341      return new soapval('return', 'string', $session_id);
 342  }

 344  function get_output($output_handle)
 345  {
 346      // go to $output_handle directory
 347      if (! chdir("$local_tmp_path/$output_handle")) {
 348          return new soap_fault('SERVER', '',
 349              "Specified job $output_handle does not exist", '');
 350      }
 351  
 352      $status = grock_get_status();
 353      if ($status == FALSE) {
 354          // raise an exception
 355          return new soap_fault('SERVER', '', 
 356          "Cannot find status of your job $output_handle", '');
 357      }
 358      if ($status == 'aborted') {
 359          // raise an exception
 360          return new soap_fault('SERVER', '', 
 361          "Job $output_handle has been aborted", '');
 362      }
 363      $status = trim($status);
 364      $output = file_get_contents("./grock_output");
 365      $error = file_get_contents("./grock_error");
 366  
 367      // have some default values for not-common vars
 368      $report = '';
 369      $failedJobs = array();
 370      
 371      if ($status == 'submitting') {
 372          $report = file_get_contents('./target_list.txt');
 373      }
 374  
 375      if ($status = 'partial') {
 376          $report = file_get_contents('./top_scores.txt');
 377      }
 378      
 379      if ($status = 'finished') {
 380          $report = file_get_contents('./top_scores.txt');
 381          // get list of failed jobs
 382          $unfinished_jobs = 0; 
 383          $total_jobs = 0;
 384          while (!feof($target_list)) {
 385              $line = fgets($target_list);
 386              if ($line == "") continue;
 387              $target = strtok(trim($line), ' ');
 388              $full_desc = strtok("\n");
 389          $total_jobs++;
 390  
 391          if (!is_dir("$target/job_output")) {
 392                  // this is a failed job
 393                  $unfinished_jobs++;
 394  #                $failedJobs[$target] = $full_desc;
 395              $failedJobs .= $line;
 396          }
 397          }
 398      }
 399      return array(
 400          'status' => $status,
 401          'output' => $output,
 402          'error' => $error,
 403          'report' => $report,
 404          'failedJobs' => $failedJobs
 405      );
 406  }
