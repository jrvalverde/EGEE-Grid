 110  /**
 111   * print an error message and exit
 112   *
 113   * Whenever we detect something wrong, we must tell the user. This function
 114   * will take an error message as its argument, format it suitably and
 115   * spit it out.
 116   *
 117   * @note This might look nicer using javascript to pop up a nice window with
 118   * the error message. Style sheets would be nice too.
 119   *
 120   * @param where the name of the caller routine or the process where the
 121   *         error occurred
 122   * @param what  a description of the abnormal condition that triggered
 123   *          the error
 124   */
 125  
 126  function error($where, $what)
 127  {
 128      // format the message
 129      echo "<p></p><center><table border=\"2\">\n";
 130      echo "<tr><td><center><font color=\"red\"><strong>\n";
 131  
 132      // output the message
 133      echo "ERROR - HORROR\n";
 134  
 135      // close format
 136      echo "</strong></font></center></td></tr>\n";
 137      echo "<tr><td><center><b>$where</b></center></td></tr>\n";
 138      echo "<tr><td><center>$what</center></td></tr>\n";
 139      echo "</table></center><p></p>\n";

 142  /**
 143   * print a letal error message and die

 145   * This function is called whenever a letal error (one that prevents
 146   * further processing) is detected. The function will spit out an
 147   * error message, close the page and exit the program.
 148   * It should seldomly be used, since it may potentially disrupt the

 151   * Actually it is a wrapper for error + terminate.

 158   * @param where the name of the caller routine or the process where the

 160   * @param what  a description of the abnormal condition that triggered

 167          exit();

 207  /**
 208   *  Generate a session directory to be used for doing all our work
 209   *
 210   *  We will create a session directory on the system scratch space
 211   * to perform all our subsequent work.
 212   *
 213   *  We need to know the systemwide scratch/tmp directory (which is a
 214   * global defined in the 'config.php' file).
 215   *
 216   *  We do all our work on the system scratch space, but to avoid clashes
 217   * between simultaneous instances of this same service, we generate a
 218   * unique session_ID and use it to ensure that we are using a namespace that
 219   * is not being used by anyone else.
 220   *
 221   *  To generate the unique name, we use the fact that mkdir(2) should be
 222   * atomic and return an error (EEXIST) if the specified pathname already
 223   * exists.
 224   *
 225   *  To avoid an infinite loop in the event of an error, we allow a maximum
 226   * number of tries.
 227   *
 228   *  @return string a unique name for the sandbox. This name should be
 229   *              unique or we risk colliding with other simultaneous
 230   *              runs.
 231   */
 232  function activate_new_sandbox()
 233  {
 234      global $local_tmp_path;
 235      global $debug;
 236  
 237      if ($debug)
 238          echo "<h3>activating a new sandbox</h3>";
 239  
 240      $i = 0;
 241      do {
 242          if ($i > 10) 
 243          letal("Activate a new sandbox", 
 244              "Could not create a sandbox on $local_tmp_path after 10 tries\n");
 245          $i++;
 246          $session_id = random_number();
 247      echo "<p>$i: Trying $session_id</p>\n";
 248      } while (!mkdir("$local_tmp_path/$session_id", 0700));
 249      
 250      chdir("$local_tmp_path/$session_id");
 251      if ($debug) {
 252          $cwd = getcwd();
 253      echo "<p>Working on $cwd (session is $session_id)</p>\n";
 254      }
 255  
 256      return $session_id;
 257  }

 259  /**
 260   *  Upload the probe molecule specified by the user to the session directory
 261   * as 'probe.pdb'.
 262   *
 263   *  This function accesses the global variable $_FILES with the known name
 264   * for the user file option to upload the user data. If the user data name
 265   * is changed in the form, it should be also changed here.
 266   *
 267   *  To locate the session directory we need a global variable to tell us
 268   * which is the default work directory, and a session_id to identify a
 269   * unique session directory within it to store safely the file without
 270   * colliding with other simultaneous instances.
 271   *
 272   *  @return string containing the original name of the file uploaded if all goes well,
 273   *      otherwise die after printing an error message.
 274   *
 275   */
 276  function upload_probe_data()
 277  {
 278      global $debug;
 279      global $local_tmp_path;
 280      
 281      $dir = getcwd(); // use to bring it to the current directory
 282      
 283      if ($debug) echo "<p>Uploading user data to ".$dir."/probe.pdb</p>\n";
 284  
 285      // We want to know the name of the file uploaded. For commodity (to us)
 286      //    we'll use a supplied name. Would probably be better to use something
 287      //    else (the original name) and return it to the caller.
 288      //$uploadfile = getcwd() . basename($_FILES['upload']['name'][0]); 
 289      $uploadfile = $dir.'/probe.pdb';
 290  
 291      $probe_file = $_FILES['upload']['tmp_name'][0];
 292      $probe_filename = $_FILES['upload']['name'][0];
 293      $probe_filesize = $_FILES['upload']['size'][0];
 294      
 295      if ($probe_file == 'none' || $probe_file = '')
 296      {    
 297          // letal prints an error and dies
 298          letal("Get_probe", "No file uploaded");
 299      }
 300      
 301      if ($probe_filesize == 0)
 302      {
 303          letal("Get_probe", "Uploaded file has zero length");
 304      }
 305  
 306      // seems OK, get it
 307      if (move_uploaded_file($_FILES['upload']['tmp_name'][0], $uploadfile)) 
 308      {
 309          return basename($_FILES['upload']['name'][0]);    // original user file name
 310      } 
 311      else 
 312      {
 313          letal("Get probe", "Could not upload your probe molecule");
 314      }
 315  }
