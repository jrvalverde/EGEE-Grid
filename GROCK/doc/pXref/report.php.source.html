  35  require_once ("grock_lib.php");    // for grock_get_status()
  36  
  37  $debug=TRUE;
  38  
  39  $id=$_GET['id'];
  40  $probe=$_GET['probe'];
  41  $pt=$_GET['pt'];
  42  $db=$_GET['db'];
  43  $docker=$_GET['docker'];        // needed for program-specific output display
  44  $from=$_GET['from'];
  45  $to=$_GET['to'];
  46  
  47  // NOTE:
  48  // $from and $to are special: they may be omitted from the URL:
  49  //  $from and $to are only needed when we have results to show,
  50  //  i.e. on show_partial_results() and show_complete_results()
  51  //  which default to showing results 1-50 as a safeguard
  52  // We will check them anyway and set them to the defaults if
  53  //  not present so we may call the two mentioned functions
  54  //  properly.
  55  // This is needed so we may re-invoke ourselves for following up
  56  //  an ongoing job while at the same time allowing users to
  57  //  navigate through partial results.
  58  // The choice of numbers, which is harcoded here in a number of
  59  //  places (sic, sic, SIC!) is based on the presumption that users
  60  //  won't be interested in huge listings while at the same time
  61  //  their interest will probably concentrate on the top 50-100
  62  //  best matches.
  63  if (!isset($from))
  64      $from = 0;
  65  if (!isset($to))
  66      $to = 50;
  67  
  68  if ((strcmp($docker, '') == 0) || (strcmp($id, '') == 0) || (strcmp($probe, '') == 0) || (strcmp($db, '') == 0)) {
  69      letal("GROCK results", "Which results do you want me to show?");  
  70  }
  71  
  72  // go there
  73  if (! chdir("$local_tmp_path/$id")) {
  74      letal("GROCK results", "The session you specified does not exist!!!");
  75  }
  76  // from now on we work on the session directory
  77  
  78  
  79  // decide what to do
  80  $status = grock_get_status();
  81  
  82  if ($status == FALSE) {
  83      // special case:
  84      // the status file does not exist
  85      //    Probably GROCK was aborted and/or cleaned
  86      //    Something went airy with it. Maybe there
  87      // is something we can show
  88      //    Requires some thinking
  89      //      Should we kill GROCK and clean-up?
  90      //        Should we try to show anything left?
  91      //        should we wait and retry?
  92      //        Should this be the default case?
  93      letal('GROCK results', "Cannot find status of your job $id");
  94  }
  95  else 
  96      $status = trim($status);
  97  
  98  // TO CHANGE: this is a manual switch...case
  99  if (strcmp($status, 'starting') == 0) {
 100      show_submission_log($id, $probe, $pt, $db, $docker, $from, $to);
 101  }
 102  else if (strcmp($status, 'started') == 0) {
 103      // copying jobs over to remote GridUI
 104      show_submission_log($id, $probe, $pt, $db, $docker, $from, $to);
 105  }
 106  else if (strcmp($status, 'submitting') == 0) {
 107      show_submitting_page($id, $probe, $pt, $db, $docker, $from, $to);
 108  }
 109  else if (strcmp($status, 'submitted') == 0) {
 110      // sent, none finished yet
 111      show_wait_page($id, $probe, $pt, $db, $docker, $from, $to);
 112  }
 113  else if (strcmp($status, 'partial') == 0) {
 114      // we have started getting results
 115      show_partial_results($id, $probe, $pt, $db, $docker, $from, $to);
 116  }
 117  else if (strcmp($status, 'finished') == 0) {
 118      // OK, we are done
 119      show_results($id, $probe, $pt, $db, $docker, $from, $to);
 120  }
 121  else if (strcmp($status, 'aborted') == 0) {
 122      // Something went wrong, terribly wrong
 123      show_abort($id, $probe, $pt, $db, $docker, $from, $to);
 124  }
 125  else {
 126      // unknown status
 127      show_abort($id, $probe, $pt, $db, $docker, $from, $to);
 128      letal('GROCK results', "Unknown status of your job $id ($status)");
 129  }

 208  /**
 209   *  Show submission progression
 210   *
 211   *  This function is called when GROCK has started submitting jobs to
 212   * the grid, but has not finished yet submitting all of them.
 213   *
 214   *  To allow for persistence, GROCK logs its submission progress report
 215   * to a file in the current session directory. We must be there (in the
 216   * session directory) when being called, so we can open the GROCK report
 217   * and pass it along to the user so s/he may monitor its progress.
 218   *
 219   *  If the submission has any problem, then GROCK won't procceed to 
 220   * the 'submitted' stage, and this will allow us to see what happened
 221   * during the submission process (and hopefully what went wrong).
 222   *
 223   *  Since the submission is an ongoing process and will be followed by 
 224   * further processing, we want to monitor it on an ongoing basis: we
 225   * just resubmit ourselves every so often and notify the user so they
 226   * may update the report manually as well or make a bookmark.
 227   *
 228   *  @param string $id    The Job ID (to be used in the output report and to back-link
 229   *      to ourselves for auto-updating)
 230   *  @param string $probe The name of the probe molecule submitted by the user
 231   *  @param string $pt    How to treat the probe molecule (ligand or receptor)
 232   *  @param string $db The name of the database being screened
 233   *  @param string $docker The program used to perform the match (used to process
 234   *      job output)
 235   *  @param integer $from Top result at which display should start
 236   *  @param integer $to Top result at which display should end
 237   */
 238  function show_submission_log($id, $probe, $pt, $db, $docker, $from=1, $to=50) {
 239      global $www_app_dir;
 240  
 241      // unneeded as of now since this page is only called once submission
 242      // has finished. We will need it though very soon.
 243      auto_update('Preparing jobs for submission', 20,
 244                  $id, $probe, $pt, $db, $docker, $from, $to);
 245      echo "<body bgcolor=\"white\" background=\"$www_app_dir/images/background.png\">\n".
 246           "<center>\n<h1>GROCK results</h1>\n" .
 247       "<table  border=\"0\">\n<tr>\n<td><img src=\"$www_app_dir/images/Rodin_Penseur.jpg\"></td>\n".
 248       "<td>".
 249       "<h2>Submitting docking jobs to the Grid:</h2>\n" .
 250       "<H3><A HREF=\"$www_app_dir/script/report.php?".
 251                     "id=$id&probe=$probe&pt=$pt&db=$db&docker=$docker".
 252                 "&from=$from&to=$to\" ".
 253       "    style=\"color:blue; text-decoration:none\">Reload</A> ".
 254       "    this page to update your job status</H3>\n" .
 255       "<H3>or</H3>\n" .
 256           "<H3>Bookmark this page to access it later with [CTRL]+[D]</H3>\n" .
 257       "</td></tr></table>\n".
 258       "<p><STRONG>Note:</STRONG> ".
 259       "This page updates itself automatically every 20 seconds</p>\n" .
 260       "<table border=\"2\" bgcolor=\"#eeeeee\">\n";
 261       
 262  //    echo file_get_contents("./target_list.txt");
 263      clearstatcache();
 264      if (! file_exists('target_list.txt')) {
 265          error('Show submission log', 'No jobs generated');
 266      exit;
 267      }
 268      
 269      // It would be better to output a number of submitted jobs
 270      // and just the last 20 or so as a (scrolling) list
 271      $log = fopen('target_list.txt', 'r');
 272      if ($log) {
 273      while (!feof($log)) {
 274          // NOTE that here we take target+description together!!!
 275              $target_desc = trim(fgets($log), "\n");
 276          if ($target_desc == '') continue;
 277          echo "<tr><td>Preparing job to match <strong>$probe</strong> ".
 278               "against <strong><font color=\"green\">$target_desc</font></strong> ".
 279               "using <strong><font color=\"blue\">$docker</strong>...</td>".
 280           "<td>done</td</tr>\n";
 281      }
 282      fclose ($log);
 283      }
 284      else
 285          echo "No jobs have been prepared yet";
 286      
 287      echo "</tr>\n</table>\n</center>\n</body>\n</html>\n";
 288      return;
 289  }
