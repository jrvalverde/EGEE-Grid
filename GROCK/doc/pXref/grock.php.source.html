  67  /**
  68   *  Grock
  69   *
  70   *  Will work on the current directory
  71   *
  72   *  Expects:
  73   * -      a probe molecule named 'probe.pdb' in the CWD
  74   * -    a series of parameters in a file named 'options'
  75   *
  76   *  The 'options' file must be in the current directory, and contain
  77   * a series of option=value pairs, one on each line, divided in
  78   * two sections:
  79   * - first comes the authorization section containing details needed
  80   * to log in and activate the grid
  81   * - an empty line separates both sections
  82   * - last come the program actual options: which database to search,
  83   * docker to use and dosck-specific parameters.
  84   *
  85   *  See processor.php for more details.
  86   *
  87   *  Once the options have been processed the file is deleted. NOTE that
  88   * since the 'options' file contains user authentication data, it should
  89   * be readable only by the user running GROCK and is anyway very sensitive.
  90   * For this reason, processor.php uses a pipe instead of an actual file,
  91   * and GROCK deletes the file as soon as it is done.
  92   *
  93   *  We might do without a file getting all in the command line, but this
  94   * would be much worst as any user would then access the auth info by
  95   * issuing a ps(1) while grock is running.
  96   *
  97   *  We might read from stdin but then we would have a problem running in
  98   * the background from PHP (sic).
  99   *
 100   *  After all options have been processed we call grock_submit() to
 101   * generate all the jobs needed (one per database entry) and submit
 102   * them to the Grid.
 103   *
 104   *  Once the jobs have been sent, we enter a timed loop to collect the
 105   * job output. If we can not collect all job output back before the
 106   * predefined timeout, unfinished jobs will be ignored.
 107   *
 108   *  Each iteration we'll reap all finished jobs output and recompute the
 109   * sorted list of best scores so the user may follow the progression of 
 110   * the search on the fly.
 111   *
 112   *  GROCK will generate a number of directories and files in the current
 113   * directory. Since these may be way too many, it is advisable to run
 114   * GROCK in a separate directory that is empty except for the 'probe.pdb'
 115   * and 'options' file to start with.
 116   *
 117   *  In the initial phase, GROCK will generate a directory for each docking job
 118   * to be submitted, i.e. one for each target molecule in the database, named
 119   * after the target entry name. Each directory will contain all the material
 120   * related to docking the probe and target molecules.
 121   *
 122   *  Additionally GROCK will generate a listing of target molecules being
 123   * explored, a status file to state the processing step it is at, and two
 124   * auxiliary trace files, one for normal output and a second for 'error'
 125   * or 'abnormal' output, which may be helpful in case of need to diagnose
 126   * a failure.
 127   */
 128  function grock()
 129  {
 130      global $debug;
 131      global $output;
 132      global $app_dir;
 133      
 134      // work on the current directory: it will become the session
 135      // directory by default
 136      $session_id = basename(getcwd());
 137      
 138      // XXX JR XXX
 139      //    we might as well get all parameters in one line as if it where
 140      //    a URL and parse them with parse_str();
 141  
 142      // First get run parameters from options file
 143      //    We might use $options = STDIN, but this way we may be run in the
 144      //    background and delete the options file ASAP
 145      // start with auth data
 146      $opt_file=fopen('options', 'r'); 
 147      if ($opt_file == FALSE) 
 148          grock_letal('GROCK', 'Nothing to do!');
 149      $opt='auth';
 150      while (! feof($opt_file)) {
 151          $inline = trim(fgets($opt_file), "\r\n");
 152      // an empty line signals the start of options data
 153      if ($inline == '') {
 154          $opt='options';
 155          continue;
 156      }
 157      fwrite($output, "\n$inline");
 158      
 159      // format is param=value
 160      $optval = explode('=', $inline, 2);
 161      ${$opt}[$optval[0]] = $optval[1];
 162      }
 163      fclose($opt_file);
 164      
 165      unlink('options');
 166      
 167      if ($debug) {
 168          fwrite($output, "\nsession is $session_id\n");
 169      fwrite($output, "called from ". basename(getcwd()) . "\n");
 170          //fwrite($output, print_r($auth,TRUE));
 171          fwrite($output, print_r($options, TRUE));
 172      }
 173      
 174      // submit all jobs
 175      grock_submit($session_id, $options, $auth);
 176      
 177      // try during at most 4 days (24 * 4) = 96
 178      // XXX JR XXX
 179      //        This should be data- docker- and database-specific
 180      //        e.g.:
 181      //            database_njobs 
 182      //            * docker_maxjoblen($probe_size) 
 183      //            * 1.20                  ### allow for 20% resubmitted jobs
 184      //            / expected number of working nodes
 185      //        Note, whith docker_maxjoblen() we may tune the JDL file for timeouts
 186      //    Currently, 4 days works for a small compound against PDB40.
 187      //    With current code, it would require 40 days for a protein against PDB40
 188      //    We NEED to have data stored on remote SE's. THIS IS A MUST.
 189      //
 190      for ($i = 0; $i < 960; $i++) {
 191          sleep(3600);     // 3600: wait 1 hour before checking results again
 192      if ($debug) fwrite($output, "Checking job output availability\n");
 193          $done = grock_get_output($session_id, $auth);
 194  
 195          // generate scores file(s)
 196          dock_generate_scores($options['docker']);
 197      
 198      if ($done)
 199          break;
 200      grock_set_status('partial');
 201      }
 202      grock_set_status('finished');
 203  
 204      exit;
 205  }
