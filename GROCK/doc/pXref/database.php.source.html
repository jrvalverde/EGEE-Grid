  37  /**
  38   *  Get the next available entry from a database
  39   *
  40   *  Given a database, this routine will fetch and return the next
  41   * entry as a file named 'target.XXX' in the local directory. The
  42   * XXX extension is database dependent (e.g. 'pdb' for PDB based
  43   * datasets). Additional descriptive information regarding the entry
  44   * extracted is returned in the $entry and $description parameters
  45   * as a convenience for further use. Upon successful completion, the
  46   * routine will return TRUE.
  47   *
  48   *  If the routine can not fetch any additional entry from the database
  49   * (e.g. because it has already returned all existing entries), then
  50   * FALSE will be returned instead.
  51   *
  52   *  This function is actually a jump table entry point to the actual
  53   * routines that carry out the work. As such it acts as the plug-in
  54   * socket for database specific routines.
  55   *
  56   *  To add support for a new database, simply follow the example.
  57   * Basically, you must detect the database type from the $db parameter
  58   * and then call the appropriate routine with the same parameters.
  59   *
  60   *  Your routine should follow the footprint of this one (recommended)
  61   * and be defined in a separate, database-dependent module that is included
  62   * at the beginning of this file.
  63   *
  64   *  @param string $db        The database type/name
  65   *  @param string $db_file  The name of the actual database file
  66   *  @param string $entry    A reference parameter where we can store the name
  67   *                          of the entry returned
  68   *  @param string $description A short description (one liner) of the entry
  69   *                          contents
  70   *
  71   *  @return boolean TRUE on success, FALSE on error or end of data
  72   */
  73  function db_get_entry($db, $db_file, &$entry, &$description)
  74  {
  75      if (strcmp(substr($database, 0, 3), 'pdb') == 0) {
  76          // 'Tis a PDB database

  78      }
  79      else if (strcmp($database, 'hic-up')) {
  80          return;
  81      }
  82      else if (strcmp(substr($database, 0, 4), 'zinc') == 0) {
  83          // Zinc-derived
  84      return;
  85      }
  86  }

  89  /**
  90   *  Extract PDB entry into a file and get entry name and description
  91   *  updating a listing file for progress monitoring.
  92   *
  93   * NOTE: IT SHOULD PROBABLY
  94   *  return EODB when there are no more entries, FALSE if the current
  95   * entry failed to be retrieved, TRUE if all is well.
  96   *
  97   *  This way we would be able to skip over wrong entries... but do we
  98   * want? How would the user know? Wouldn't it be better if we were told?
  99   * At any rate, as it currently is, if an entry is wrong we stop but don't
 100   * know, meaning the user is led to believe the whole database was
 101   *
 102   *  @param string $db        The database type/name
 103   *  @param string $db_file  The name of the actual database file
 104   *  @param string $entry    A reference parameter where we can store the name
 105   *                          of the entry returned
 106   *  @param string $description A short description (one liner) of the entry
 107   *                          contents
 108   *
 109   *  @return boolean TRUE on success, FALSE on error or end of data
 110   */
 111  function PDB_get_entry($db, $db_file, &$entry, &$description)
 112  {
 113      global $debug;
 114      global $pdb_path;
 115      global $mmtsb_tools;
 116      global $output;
 117      global $db_dir;
 118      global $dblist;         // we'd like it static!
 119      global $targetlist;
 120      
 121      $entry='';                // Clear data
 122      $database='';
 123  
 124      if ($debug) fwrite($output, "PDB_get_next_entry() called\n");
 125  
 126      if ($dblist == FALSE) {
 127          // if not already open, open it for reading
 128          $dblist = fopen("$db_dir/$db.lst", "r");
 129      if (!$dblist) {
 130          echo_error('PDB get entry', "Cannot open database $db_dir/$db.lst");
 131          exit;
 132          return FALSE;
 133      }
 134      }
 135      
 136      if ($targetlist == FALSE) {
 137          // if not already open, open it for writing
 138      $targetlist = fopen("./target_list.txt", "w");
 139      if (! $targetlist) {
 140          echo_error('PDB get entry', "Cannot open listing target_list.txt");
 141          exit;
 142          return FALSE;
 143      }
 144      }
 145      
 146      // get next entry line (skip blank lines)
 147      do {
 148          if (feof($dblist)) {
 149          fclose($dblist);
 150          fclose($targetlist);
 151          return FALSE;
 152      }
 153          $line = trim(fgets($dblist));
 154      } while ($line == '');
 155  
 156      // parse it
 157      $entry = rtrim(substr("$line", 5, 6));     // ENTR_{C} (entry_chain)
 158      $description = substr($line, 11);            // description
 159  
 160      // $target is ENTR_C (entry_chain)
 161      //      entry is the first four letters in lower case
 162      //        entry filename is $pdb_dir/pdb${entry}.ent
 163      //        chain is the sixth letter if any
 164      $entry_file="pdb".strtolower( substr($entry, 0, 4) ).".ent";
 165      $chain=substr($entry, 5, 1);
 166      if ($debug) fwrite($output, "\nProcessing $entry ($entry_file - $chain)\n");
 167      
 168      // create the target database file
 169      if ($chain=="") {
 170          // if there is no special chain to use, then use
 171          // the full entry
 172      if ($debug) {
 173          fwrite($output, "copying $pdb_path/$entry_file to ./$db_file\n");
 174              $exit = 0;
 175      }
 176      exec("cp $pdb_path/$entry_file ./$db_file", $out, $exit);
 177      if ($exit != 0) {
 178          echo_error("Get PDB entry", "couldn't get $entry");
 179          fwrite($output, print_r($out, TRUE));
 180          return FALSE;
 181      }
 182      } else { 
 183          // extract chain into output file
 184      if ($debug) {
 185          fwrite($output, "extracting chain $chain from $pdb_path/$entry_file into ./$db_file\n");
 186          $exit = 0;
 187      }
 188      exec("$mmtsb_tools/convpdb.pl -chain $chain $pdb_path/$entry_file > ./$db_file", $out, $exit);
 189      if ($exit != 0) {
 190          echo_error("Get PDB entry", "couldn't create $db_file from $entry_file");
 191          fwrite($output, print_r($out, TRUE));
 192          return FALSE;
 193      }
 194      }
 195      unset($out);
 196      
 197      if ($debug) {
 198          exec("ls -l ./$db_file", $out);
 199      fwrite($output, print_r($out, TRUE));
 200          fwrite($output, "\n");
 201      }
 202  
 203      // Update listing of processed entries
 204      // This is needed only for tracking the job preparation process
 205      //    although it is used as a convenience for processing listings
 206      //    The format of target_list.txt is simple: each line contains
 207      //    an entry followed by its description.
 208      if (fwrite($targetlist, "$entry $description\n") == FALSE) {
 209      // if ($exit != 0) {
 210      echo_error("Get PDB entry", "couldn't add $entry to list file");
 211      fwrite($output, print_r($out, TRUE));
 212      return FALSE;
 213      }
 214      return TRUE;
 215  }
