   1  #!/bin/sh
   2  #
   3  #    get-results.sh directory
   4  #
   5  #    Extracts the results from a grid tar file inside a job directory.
   6  # This is designed to be used inside GROCK (GRid dOCK).
   7  #
   8  #    The problem: we get all the results as independent files for each
   9  # job inside separate directories, one for each job. Each directory/job
  10  # contains the I/O data for a different dataset. Output is stored as a
  11  # gzipped tarball in a subdirectory. All result files follow the same
  12  # format, which does NOT include info about the original dataset. What
  13  # we want is to extract the results and tag them with info regarding the
  14  # original data set.
  15  #
  16  #    Now: each dataset is composed of two files, one is constant along
  17  # ALL the jobs (the probe molecule) while the other one is the different
  18  # element that distinguishes each job. The directory holding the results
  19  # for each job is named after this distinguishing component.
  20  #
  21  #    Hence: we extract the results file from the tarball in the given
  22  # directory and use the directory name (same as distinguishing element in
  23  # the data set) to tag every line of the results.
  24  #
  25  #    We tag at the end of the lines for one major reason: the component
  26  # name has no standard length, and thusly it is less disruptive. It is also
  27  # easier to select e.g. the highest scoring match, by grep'ping for '^   1'
  28  #
  29  #    Usage (simple):
  30  #        get-results.sh COMPOUND
  31  #
  32  #    where COMPOUND is the name of the directory holding the results
  33  #
  34  #    USAGE (FULL):
  35  #    TO GATHER ALL RESULTS AND SORT THEM BY ENERGY
  36  #        cd $output_dir
  37  #        # where $output_dir contains all the results collected
  38  #        # each in its own subdir named after the distinguishing
  39  #        # compound
  40  #        ls | xargs -i -t get-results.sh {} | sort -r -n -k 2
  41  #
  42  #
  43  # Initial, raw metod:
  44  #  find . -name gramm-come.tar.gz -exec tar -zOxvf {} receptor-ligand.res \; |\
  45  #         sort -r -n -k 2 > sorted.res
  46  #
  47  # Now, this has a problem: it does not save the receptor name and we need it.
  48  # For this we need to know the receptor name...
  49  #
  50  # Let us go for xargs:
  51  # 
  52  # ls | xargs -i -t get-results.sh {} | sort -r -n -k 2
  53  #
  54  #    -i : go line by line and substitute {} by name
  55  #    -t : print to stderr each processed name (useful for debugging and
  56  #         tracing

  84  if [ -f $1/job_output/gramm-come.tar.gz ] ; then
  85      tar -zxvOf $1/job_output/gramm-come.tar.gz receptor-ligand.res | \
  86          tail -n +32 | \
  87          sed -e "s/\$/  $1/g" 
  88      # fix lack of \n on last line of results file.
  89      echo ""
  90  fi
  91  exit 1
