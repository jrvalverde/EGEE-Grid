  52  /**
  53   *  Submit a high throughput docking search using a remote grid
  54   *  back-end.
  55   *
  56   *  We expect to be called with a session name that identifies
  57   * a directory in the working area which already contains the
  58   * probe molecule file as 'probe.pdb'.
  59   *
  60   *  The parameters allow us to
  61   *  - know the original name of the probe file
  62   *  - select the database to be searched
  63   *  - select the docking program to use and all its options
  64   *  - identify ourselves to the remote back-end
  65   *
  66   *  With this data in hand, we will be able to generate all
  67   * the separate jobs for performing the search with as much
  68   * parallelism as possible, and launch them to a remote
  69   * back end for processing.
  70   *
  71   *  In doing so, we will maintain a file with our status so
  72   * work may be monitored by an external process, a listing
  73   * file detailing all target molecules checked and a log file
  74   * in HTML format with a progress report (this last one may
  75   * be superfluous).
  76   *
  77   *  $options is an associative array containing
  78   *      'probe'
  79   *      'database'
  80   *      'docker' => 'gramm'
  81   *          'resolution'
  82   *          'representation'
  83   *
  84   *  $auth is an associative array containing
  85   *      'server'
  86   *      'user'
  87   *      'host'
  88   *      'password'
  89   *      'passphrase'
  90   *
  91   *  @param string $session_id    Session identifier
  92   *  @param mixed $options       An array containing the user options
  93   *  @param mixed $auth            An array containing user authentication data
  94   */ 
  95  function grock_submit($session_id, $options, $auth)
  96  {       
  97      // we use globals to retrieve configuration file values
  98      global $db_dir;
  99      global $debug;
 100      global $output;
 101  
 102      if ($debug) fwrite($output, "Running GROCK\n-------------\n\n");
 103      // for clarity
 104      $probe = $options['probe'];
 105      $database = $options['database'];
 106  
 107      grock_set_status('started');
 108      
 109      // Get data files
 110      //  We have two cases here:
 111      //        1- If the user selected a PDB database then his probe
 112      //        data will be mobile during docking, and the database
 113      //        macromolecules will be static. We want the user data
 114      //        uploaded to $session_directory/ligand.pdb
 115      //
 116      //        2- If the user selected a compound database, then
 117      //        his probe data is a macromolecule that must remain fixed
 118      //        and should be uploaded as $session_directory/target.pdb
 119      //        while thedatabase compounds whill be mobile and stored
 120      //        on ligand.pdb
 121      if (strcmp(substr($database, 0, 3), 'pdb') == 0) {
 122          //$probe_type = 'ligand';
 123          $user_file = 'ligand.pdb';      // name to assign the user probe file
 124      $db_file = 'receptor.pdb';      // name to assign the DB target file
 125      }
 126      else {
 127          //$probe_type= 'receptor';
 128          $user_file = 'receptor.pdb';    // so that docker knows how to treate
 129      $db_file = 'ligand.pdb';        // the user and db molecules.
 130      }
 131      if ($debug) fwrite($output, "\n    Your data will be $user_file and matched against $db_file\n");
 132      
 133      // ensure user data goes into correct file
 134      grock_set_probe_type($user_file);
 135      // $probe is the original user file name
 136      if ($debug) fwrite($output, "\n    Successfully moved $probe to $user_file\n");
 137  
 138      $jobs = 0;
 139      $status = 0;
 140      while (PDB_get_entry($database, $db_file, $target, $description)) {
 141      
 142          // Prepare a docking job to be sent over to the grid
 143          //
 144          //  The molecular data are already stored using always
 145          // the same file names 'ligand.pdb' and 'receptor.pdb'
 146              $jobs++;
 147          if (!dock_create_job($probe, $target, $options)) {
 148              grock_letal("GROCK submit", "Couldn't create job for $probe+$target on $session_id");
 149          }
 150          unlink($db_file);   // no longer needed
 151  
 152      }
 153  
 154      if ($jobs == 0)
 155          grock_letal("GROCK submit", "Couldn't get any molecule to search");
 156  
 157      if ($debug) {
 158          fwrite($output, "\n    Contents of $session_id\n");
 159          exec("ls -l *", $out);
 160          fwrite($output, print_r($out, TRUE));
 161          fwrite($output, "\n");
 162      }
 163  
 164      // OK, we are ready, let's go for the jugular
 165      // we need to pass $session_id for the remote back-end, do we?
 166      submit_session_jobs($session_id, $auth);
 167  
 168  
 169      // The signal to notice the job is done is the "jobs_sent.txt" file
 170      grock_set_status('submitted');
 171  }

 174  /**
 175   *  Ensure the user probe molecule has the appropriate name for further
 176   * processing by GROCK.
 177   *
 178   *  GROCK works internally using 'canonical' names for all its processing
 179   * (this makes the logic somewhat simpler, although it will probably be
 180   * removed at some future point).
 181   *
 182   *  This routine expects the user molecule to be located in the current
 183   * directory, named 'probe.pdb', and moves it to the actual name that
 184   * GROCK will need to process it.
 185   *
 186   *  The need to rename stems from the fact that the molecule will be 
 187   * processed as a ligand (mobile) or receptor (static) molecule depending
 188   * on its size and the database selected. Hence we want to make sure it
 189   * is identified appropriately for the processing to be applied.
 190   *
 191   *  @access private
 192   *
 193   *  @param string $user_file the name to assign the newly uploaded data file.
 194   *
 195   */
 196  function grock_set_probe_type($user_file)
 197  {
 198      global $debug;
 199      global $output;
 200      
 201      if ($debug) fwrite($output, "Moving user data to ".getcwd()."/$user_file\n");
 202  
 203      clearstatcache(); 
 204      if (! file_exists('./probe.pdb')) {
 205          grock_letal("Get probe data", "The coordinates of the probe molecule are not available!");
 206      }
 207      if (rename('./probe.pdb', $user_file)) 
 208      {
 209          return TRUE;
 210      } 
 211      else 
 212      {
 213          grock_letal("Get probe data", "Could not access the coordinates of your probe molecule");
 214      }
 215  }

 263  /**
 264   *  Submit all session jobs at once to the remote back-end.
 265   *
 266   *  This function expects a session_id which identifies a directory
 267   * holding a collection of jobs to be submitted on the local temporary
 268   * work directory.
 269   *
 270   *  All jobs are sent to the remote back-end, stored there on the remote
 271   * temporary work directory and then launched for execution on the grid.
 272   *
 273   *  Since we are to use a remote back-end connected to the Grid, we need
 274   * to know all the authentication details:
 275   *      $auth is an array that holds
 276   *          'server' => user@server identifier
 277   *          'password' => password to use to connect to 'server'
 278   *          'passphrase' => passphrase needed to unlock the Grid from 'server'
 279   *
 280   *  In order to convert a session_id into a directory we also need to access
 281   * the GLOBALS
 282   *      $local_tmp_path
 283   *      $gridUI_tmp_path
 284   *
 285   *  As a SIDE EFFECT, the LCG-tools will be installed on the remote temporary
 286   * work directory so they may be used for submission, AND BE LEFT there (so
 287   * they might be shared by other instances).
 288   *
 289   *  If anything goes wrong, clean up local and remote directories and die.
 290   *
 291   *  XXX JR XXX NOTE: reverse-engineer lcg-submitter and rewrite using PHP
 292   *
 293   *  @param  string $session_id     The identifier for our current session
 294   *  @param  mixed  $auth           an array holding authentication information
 295   *
 296   *  @return void if everything OK, die with an error otherwise.
 297   */
 298  function submit_session_jobs($session_id, $auth)
 299  {
 300      global $debug;
 301      global $debug_sexec;
 302      global $local_tmp_path;
 303      global $gridUI_tmp_path;
 304      global $app_dir;            // to locate LCG-tools
 305      global $output;
 306   
 307      grock_set_status('submitting');
 308      if ($debug) {
 309          fwrite($output, "\nSubmitting session jobs\n\n");
 310      flush();
 311      //return;     // DO NOT SUBMIT WHILE DEBUGGING
 312      //$debug_sexec = TRUE;
 313      }
 314      
 315      $local_dir = "$local_tmp_path/$session_id";
 316      $remote_dir = "$gridUI_tmp_path/$session_id";
 317      
 318      $rmt = new SExec($auth['server'], $auth['password']);
 319      if ($rmt == FALSE) {
 320          grock_letal("Submit session jobs",
 321          "Could not connect to remote system $auth['server']}");
 322      }
 323      if ($debug) fwrite($output, "    Connected to $auth['server']}\n");
 324      
 325      // Copy the whole session directory with all the job files 
 326      if (! $rmt->ssh_copy_to($local_dir, $remote_dir, $out)) {
 327          $rmt->destruct();
 328          // XXX clean-up directories here
 329      if (! $debug)    grock_clean();
 330      if ($debug) fwrite($output, print_r($out, TRUE));
 331          grock_letal("Submit session jobs",
 332          "Could not copy jobs to remote back-end for execution");
 333      }
 334      if ($debug) fwrite($output, "    Copied session $session_id\n");
 335      
 336      // There are files that we don't need in the remote directory, 
 337      // ligand.pdb, receptor.pdb and pdb_list.txt
 338      $cmd = "(cd $remote_dir; rm -f grock_error grock_output ".
 339              "grock_status options ligand.pdb receptor.pdb ".
 340          "target_list.txt )";
 341      $exit = $rmt->ssh_exec("$cmd", $out);
 342      if ($exit != 0) {
 343          echo_warning("GROCK submission: Could not remove unneeded files from remote back-end");
 344      fwrite($output, "$out\n");
 345      }
 346      unset($out);
 347      if ($debug) fwrite($output, "    Removed unneeded files\n");
 348  
 349      // Install LCG tools
 350      //   We will be using auxiliary tools to launch all the jobs 
 351      //   'en masse'. Since they are not yet on the standard install,
 352      //   we must copy them over so we can use'm.
 353      //      The submitter will loop over all subdirs in $session_id,
 354      //      hence it must be placed outside $session_id.
 355      //        We may as well share them between sessions: first come
 356      //    installs the tools, followers use them.
 357      $lcg_dir = "$app_dir/lcg-tools";
 358      // is it already there? 
 359      $exit = $rmt->ssh_exec("test -x $gridUI_tmp_path/lcg-tools/lcg-submitter-biomed-beta11.pl", $out);
 360      unset($out);    // we don't care about it
 361      if ($exit != 0) {
 362          if ($debug) fwrite($output, "    Installing LCG-tools\n");
 363          // It is not already there yet. Install it as $gridUI_tmp_path/lcg-tools.
 364          if (! $rmt->ssh_copy_to($lcg_dir, "$gridUI_tmp_path/lcg-tools", $out)) {
 365              // we are toasted: clean and exit
 366          if (! $debug) {
 367              $rmt->ssh_exec("rm -rf $remote_dir", $out);
 368              grock_clean();
 369          }
 370          $rmt->destruct();
 371          if ($debug) fwrite($output, print_r($out, TRUE));
 372              grock_letal("Submit session jobs",
 373          "Couldn't make a copy of the LCG tools to $session_id<br>\n");
 374      }
 375      }
 376      
 377      // Prepare the grid for work
 378      activate_grid($session_id, $auth);
 379      
 380      // The submiter tool launches all the jobs.
 381      if ($debug) fwrite($output, "    Submitting jobs with lcg-submitter\n");
 382      $cmd = "/usr/bin/perl $gridUI_tmp_path/lcg-tools/lcg-submitter-biomed-beta11.pl -session $remote_dir";
 383      if ($debug)
 384          // save output on remote UI tmp dir for manual monitor with 'tail -f'
 385          $exit = $rmt->ssh_exec("$cmd > $gridUI_tmp_path/$session_id/submitter_output 2>&1", $out);
 386      else
 387          $exit = $rmt->ssh_exec("$cmd", $out);
 388      if ($exit != 0) {
 389      if (! $debug) {
 390          // XXX we'll leave the lcg-tools cached remotely
 391          $rmt->ssh_exec("rm -rf $remote_dir", $out);
 392          grock_clean();
 393      }
 394          $rmt->destruct();
 395      if ($debug) fwrite($output, print_r($out, TRUE));
 396          grock_letal("Submit session jobs",
 397          "An error occurred during submission of $session_id");
 398      }
 399  
 400      if ($debug) {
 401          // Stop here (move me around for debugging different sections)
 402      $rmt->destruct();
 403      fwrite($output, "\n");
 404      return;
 405      }
 406      
 407      $rmt->destruct();
 408      if ($debug) fwrite($output, "\n");
 409  }

 411  /**
 412   *  Set work progression status
 413   *
 414   *  Ours is a lengthy process, we do not want users to be tied
 415   * to a terminal waiting for this to finish. Since we are web
 416   * based, this is done by a separate web page, hence we need 
 417   * some persistent way to monitor work progress.
 418   *
 419   *  This routine isolates the persistence implementation of
 420   * progression status.
 421   *
 422   *  @param string $status   status to set
 423   *
 424   *  @return boolean  TRUE if all goes OK,
 425                      FALSE and a warning if something goes awry
 426   */
 427  function grock_set_status($status) {
 428      $sf = fopen('grock_status', 'w');
 429      if (! $sf) {
 430          echo_warning('GROCK set status: Can not create status file');
 431      return FALSE;
 432      }
 433      if (! fwrite($sf, $status)) {
 434          echo_warning('GROCK set status: Can not update status');
 435      return FALSE;
 436      }
 437      fclose($sf);
 438      flush();
 439      return TRUE;
 440  }

 442  /**
 443   *  Obtain status of GROCK process
 444   *
 445   *  As we need to provide access to a running GROCK jobs from
 446   * indpenendent instances of web pages invoked by the user at
 447   * his/her convenience, we need some way to access a presistent
 448   * instance of the progression status.
 449   *
 450   *  This routine allows us to access persistent storage hiding
 451   * implementation details.
 452   *
 453   *  @return string status of running GROCK or FALSE if the status
 454   *      could not be obtained (denotes a possible error condition).

 456  function grock_get_status()
 457  {
 458      clearstatcache(); 
 459  
 460      if (!file_exists('grock_status'))
 461          return FALSE;
 462  
 463      $sf = fopen('grock_status', "r");
 464      if (! $sf)
 465          return FALSE;
 466      $status = fgets($sf);
 467      fclose($sf);
 468      return $status;
 469  }

 514  /**
 515   *  Get output for GROCK jobs
 516   *
 517   *  This function will loop over all jobs and check their status. If
 518   * they are finished, then it will check the output code:
 519   * 
 520   * - if not OK then resubmit the job once
 521   * - if OK then get its output
 522   *
 523   *  XXX If timeout, jobs should be killed and resubmitted as well... Errr.
 524   *      How do we timeout? Maybe the JDL is the answer... But how long do
 525   *      we allow a job to run?
 526   *
 527   *  After it has finished the loop, if all output has been collected, 
 528   * then it will return TRUE.
 529   *
 530   *  If some output could not be collected because either the jobs
 531   * were not completed, they terminated with error or the output is
 532   * not available, then return FALSE
 533   *
 534   *  NOTE: that is too simplistic and it might make sense to 
 535   * indicate on return whether
 536   *  - there are incomplete jobs
 537   *  - there are error-terminated jobs
 538   *  - there are terminated jobs with no output
 539   *  - everything went OK
 540   *
 541   *  This function must be called repeatedly and periodically as
 542   * otherwise we risk a job being completed but its results having
 543   * been lost/discarded for some reason.
 544   *
 545   *  @param string $session_id the session whose jobs we will monitor
 546   *  @param array $auth the authentication information
 547   *
 548   *  @return boolean TRUE if all output has been collected,
 549   *                  FALSE otherwise
 550   */
 551  function grock_get_output($session_id, $auth)
 552  {
 553      global $output;
 554      global $debug;
 555      global $local_tmp_path;
 556      global $gridUI_tmp_path;
 557      
 558      if ($debug) 
 559          fwrite($output, "\nGrock get output\n\n    connecting to the Grid\n");
 560      $eg = new Grid;
 561      $eg = new Grid;
 562      if ($eg == FALSE) {
 563          // bad luck this time, may be next time we are called we'll be
 564      // luckier
 565           warning("Activating grid: ".
 566      "Cannot open connection to Grid back_end $grid_server!\n");
 567      return FALSE;
 568      }
 569      
 570      $eg->set_host($auth['host']);
 571      $eg->set_user($auth['user']);
 572      $eg->set_password($auth['password']);
 573      $eg->set_passphrase($auth['passphrase']);
 574      $eg->set_error_log("$local_tmp_path/$session_id/error-output.txt");
 575      $eg->set_work_dir("$gridUI_tmp_path");
 576      $eg->connect();
 577      $eg->initialize(2, 0);     // a 2h window should be enough
 578  
 579      // we are called from within $session_id, but need
 580      // to tell the Grid to access directories in the
 581      // remote corresponding session directory
 582      if ($debug) fwrite($output, "    Defining remote session as $session_id\n");
 583      $eg->session_define($session_id, $session_id);
 584      
 585      $tdb = fopen("target_list.txt", "r");
 586      if (!$tdb) {
 587          // we're toasted, we won't be able to even show any results
 588      //  XXX JR XXX at this point a recovery attempt to build the 
 589      //  target_list.txt file scanning the subdirectories might be
 590      //  in place.
 591          grock_letal("Get Results", "Cannot read the list of database molecules screened.");
 592      }
 593      $complete = TRUE;
 594      while (! feof($tdb)) {
 595          $target = strtok(trim(fgets( $tdb )), ' ');
 596      if (strcmp($target, '') == 0)
 597          continue;
 598      // check job status
 599      unset($out);
 600      // NOTE this check is here to save a remote call to job_status()
 601      if (is_dir("$target/job_output"))       // if already done
 602          continue;
 603      if ($debug) fwrite($output, "    Checking $target\n");
 604          if (!$eg->job_status($target, $out, $session_id)) {
 605          // e.g. network down.
 606          echo_warning("Couldn't get status of $target");
 607              if (job_resubmit($eg, $session_id, $target)) {
 608              // give it a chance to run
 609          $complete = FALSE;
 610                  continue;
 611          }
 612          else {
 613              // the job can't be resubmitted at this time,
 614          // since there is nothing more we can do now, ignore it
 615              continue;
 616          }
 617      }
 618      
 619      // Did it complete?
 620      if (strstr($out[6], 'Cancelled') || strstr($out[6], 'Aborted')) {
 621          // either we manually cancelled the job or we have run on a
 622          // hard limit (out of proxy or out of running time) In any
 623          // case this job will have to be ignored.
 624          continue;
 625      }
 626      if (! strstr($out[6], 'Done')) {    // No, we have to wait longer
 627          $complete = FALSE;
 628          if ($debug) fwrite($output, "    --> $target is not done yet\n");
 629      }
 630  
 631      // yes, it finished
 632      else if (strstr($out[6], 'Done (Success)')) {
 633          // this one is ready, fetch it if not got yet
 634          // NOTE: this test might go UP ABOVE and save a gridUI call
 635          if ($debug) fwrite($output, "    --> $target is done: retrieving\n");
 636          if (! is_dir("$target/job_output")) {
 637              unset($out);
 638              $eg->job_get_output($target, $out, $session_id);
 639          // we don't care for the result: if it failed, then
 640          // it will be repeated on the next iteration.
 641          }
 642      } else {
 643          // job is done but failed: resubmit once more
 644          // If it has already been resubmitted, we give up
 645          if ($debug) fwrite($output, "    --> $target failed\n");
 646              if (job_resubmit($eg, $session_id, $target)) {
 647          $complete = FALSE;      // give it a chance to run
 648          }
 649          else
 650              // the job can't be resubmitted now, so we can only
 651          // ignore it for the time being
 652              if ($debug) fwrite($output, "        Giving up $target for now\n");
 653          
 654      }
 655      }
 656      fclose($tdb);
 657      
 658      // $complete will hold true only if all jobs have reached one of either 
 659      // the 'Done', 'Cancelled' or 'Aborted' status (possibly after resubmission).
 660      //    If a job is resubmitted, then $complete will have been set to
 661      // FALSE to give it a chance to be rerun.
 662      if ($complete)
 663          if (!$debug) {
 664          $eg->session_destroy($session_id);    // clean up remote site
 665          //$eg->destroy();         // this may be dangerous as it might
 666                                  // kill other user jobs (unused yet)
 667      }
 668      $eg->destruct();
 669      return $complete;
 670  }

 672  /**
 673   *  Resubmit a job to the grid up to a globally defined maximum number of times
 674   *
 675   *  We'll use an ancillary trace file in the job directory (named 'resubmitted')
 676   * to hold a count oof the number of times the job has been resubmitted.
 677   *
 678   *  @param Grid $eg    An open, active grid connection to use
 679   *  @param string $session_id The session identifier
 680   *  @param string $job The job identifier
 681   *
 682   *  return TRUE if the job was resubmitted, FALSE otherwise.
 683   */
 684  function job_resubmit($eg, $session_id, $job)
 685  {
 686      global $debug;
 687      global $output;
 688  
 689      if ($debug) fwrite($output, "job_resubmit($eg, $session, $job)\n");
 690      $prev_tries = 0;
 691      // Have we already resubmitted the job?
 692      if (file_exists("$job/resubmitted")) {
 693          // How many times?
 694      $countf = fopen("$job/resubmitted", "r");
 695      if ($countf == FALSE) {
 696          // something smelly is going on
 697          if ($debug)  fwrite($output, "--> Cannot open existing resubmitted file\n");
 698          return FALSE;
 699      }
 700      fscanf($countf, "%d", $prev_tries);
 701      fclose($countf);
 702      }
 703      if ($debug) fwrite($output, "--> Previously tried $prev_tries times\n");
 704      
 705      // Have we exceeded the maximum count?
 706      if ($prev_tries > MAX_TRIES) {
 707          if ($debug) fwrite($output, "--> Maximum number of tries exceeded\n"); 
 708          return FALSE;
 709      }
 710      
 711      // clean up
 712      if (file_exists("$job/identifier.txt")) {
 713          unlink("$job/identifier.txt");
 714      }
 715      if (file_exists("$job/submit.txt")) {
 716          unlink("$job/submit.txt");
 717      }
 718      
 719      // submit job
 720      // NOTE that we expect a valid Grid session/object
 721      $ok = $eg->job_submit($job, $out, $session_id);
 722      if (! $ok)
 723          return FALSE;
 724  
 725      if ($debug) fwrite($output, "--> Resubmitted $job\n");
 726      // update count
 727      $countf = fopen("$job/resubmitted", "w");
 728      if ($countf == FALSE) {
 729          // something's wrong, but we have already resubmitted
 730      // the job although we can't update the count.
 731      // In a worst case the job will be re-tried more than
 732      // MAX_TRIES times, but only until grock lifespan finishes.
 733      return TRUE;
 734      }
 735      fwrite($countf, $prev_tries + 1);
 736      fclose($countf);
 737  
 738      return TRUE;
 739  
 740  }
