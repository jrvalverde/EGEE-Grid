   1  #!/bin/sh
   2  #
   3  # To test, run as 
   4  #    ssh -T localhost SSH.sh $user@host
   5  #    enter-password
   6  # this will run this script under NO TTY.
   7  #
   8  # NOTES:
   9  # 1.    You can remove the "read password" and substitute $password by $2
  10  #    Then the password will be taken from the command line
  11  #    But this is insecure as it shows up on 'ps'
  12  #
  13  # 2.    You can simplify it a lot by removing all pipe-related stuff and
  14  #    just making './catp' echo the password.
  15  #    But then the password will be stored on './catp' which is root
  16  #    readable. Do you trust your 'root'?
  17  #
  18  # 3.    There's a race condition here between the issuance of the 'echo'
  19  #    command and the reading of the pipe by 'catp'. After you write the
  20  #    password on the pipe, it will be available there while ssh starts
  21  #    and then while it calls './catp' and this in turn calls 'cat' or
  22  #    'head'. Someone might read the pipe meanwhile, get the password,
  23  #    save it and re-write it on the pipe.
  24  #    But the time window is a lot shorter than simply leaving the 
  25  #    password around in the './catp' script.
  26  #
  27  # 4.    When we are trusted, the password is ignored for login. Anything
  28  #    will do.
  29  #    But for some reason the ssh process hangs waiting on exit... this
  30  #    needs further investigation.
  31  
  32  # read password (from stdin)
  33  read password
  34  
  35  # set environment
  36  export DISPLAY=none:0.0
  37  export SSH_ASKPASS=/tmp/catp.$RANDOM}.$RANDOM}
  38  
  39  # create auxiliary command
  40  umask 077
  41  cat > $SSH_ASKPASS << END
  42  #!/bin/sh
  43  rm -f $SSH_ASKPASS
  44  echo $password    # this stores the password on this file!!!
  45  exit
  46  END
  47  chmod 700 $SSH_ASKPASS
  48  
  49  # let's go!
  50  
  51  ssh -x -t -t $1
  52  
  53  # just in case: if we are trusted the script is not run and therefore it
  54  # can't delete itself.
  55  rm -f $SSH_ASKPASS
  56  
  57  exit
  58
