  23  /**
  24   * Install the program and scripts needed to run it on a suitable place
  25   *
  26   *    This function installs the docking program GRAMM and the scripts
  27   * needed to run it on a suitable local directory.
  28   *
  29   *    Since we may potentially be called many times in a short period
  30   * we need a way to make sure their jobs don't clash with one another,
  31   * most specially, that they don't overwrite each others' workspace.
  32   *
  33   *    For this reason, we use a separate working directory for each
  34   * request. This function will install GRAMM on the specified working
  35   * directory.
  36   *
  37   *    @param wd_path    local directory devoted to Grid work
  38   *    @param random_value    a random value to generate a unique
  39   *                name for the working directory.
  40   */
  41  function install_gramm($wd_path, $random_value)
  42  {    
  43      // We copy the results.php to the working directory
  44      if (!copy( "./results.php", "$wd_path/$random_value/index.php" ))    
  45      {
  46          echo "Error, can't copy results.php to the working directory<br /></body></html>\n";
  47          exit;    
  48      } 
  49      
  50      // We copy the functions.php to the working directory
  51      if (!copy( "./functions.php", "$wd_path/$random_value/functions.php" ))    
  52      {
  53          echo "Error, can't copy functions.php to the working directory<br /></body></html>\n";
  54          exit;    
  55      } 
  56      
  57      // We copy the variables.php to the working directory
  58      if (!copy( "./variables.php", "$wd_path/$random_value/variables.php" ))    
  59      {
  60          echo "Error, can't copy variables.php to the working directory<br /></body></html>\n";
  61          exit;    
  62      } 
  63      
  64      // We copy the egee logo
  65      if (!copy( "../interface/egee.jpg", "$wd_path/$random_value/egee.jpg" ))    
  66      {
  67          echo "Error, can't copy  egee.jpg to the working directory<br /></body></html>\n";
  68          exit;    
  69      } 
  70      // We move the gramm files to the working directory
  71      // XXX JR XXX -- this should suffice
  72          exec("cp -f gramm/* $wd_path/$random_value/gramm/");
  73          // otherwise use
  74          // - in variables.php

  79  /**
  80   * Create working directory and move to it
  81   *
  82   *  The goal is to go to the working directory. If it does not
  83   * exist, we create it (it shouldn't) and move inside.
  84   *
  85   *  Ideally we would also create an .htaccess file and a .htpasswd
  86   * with a random password to return to the user. Should that be done
  87   * here?
  88   *
  89   * @note The working directory should not exist!
  90   *
  91   * @param $user_wd_path the _absolute_ path to the local directory where 
  92   *              we will be storing user data.
  93   */
  94  function go_to_work($user_wd_path, $options)
  95  {
  96      
  97      // Working directory in the local machine
  98      if (!mkdir("$user_wd_path", 0777))
  99      {
 100          echo "ERROR, HORROR: cannot generate a working directory<br /></body></html>\n";
 101          exit;
 102      }
 103      // Mkdir seems to not handle properly the permissions
 104      chmod( "$user_wd_path", 0777 );
 105  
 106          chdir("$user_wd_path");    
 107  }

 201  function create_gr_files($user_wd_path, $resolution)
 202  {
 203  // We generate rmol.gr, rpar.gr and wlist.gr, requiered to run gramm
 204  
 205  // Writing rmol.gr
 206  
 207  $rmol="$user_wd_path/tinker/rmol.gr";
 208  $receptor = "receptor";
 209  $ligand = "ligand";
 210  
 211  $fp = fopen( "$rmol", "w" );
 212  
 213      fwrite( $fp, "# Filename  Fragment  ID      Filename  Fragment  ID     [paral/anti  max.ang]\n");
 214      fwrite( $fp, "# ----------------------------------------------------------------------------\n");
 215      fwrite( $fp, "\n");
 216      fwrite( $fp, "$receptor.pdb     *    receptor  $ligand.pdb       *     ligand\n");
 217  
 218  fclose( $fp );
 219  
 220  // Writing rpar.gr
 221  $rpar="$user_wd_path/tinker/rpar.gr";
 222  
 223  $fp1 = fopen( "$rpar", "w" );
 224  
 225      if ($resolution=="low")
 226          {    
 227          fwrite( $fp1, "Matching mode (generic/helix) ....................... mmode= generic\n" );
 228          fwrite( $fp1, "Grid step ............................................. eta= 6.8\n" );
 229          fwrite( $fp1, "Repulsion (attraction is always -1) .................... ro= 6.5.\n" );
 230          fwrite( $fp1, "Attraction double range (fraction of single range) ..... fr= 0.\n" );
 231          fwrite( $fp1, "Potential range type (atom_radius, grid_step) ....... crang= grid_step\n" );
 232          fwrite( $fp1, "Projection (blackwhite, gray) ................ ....... ccti= gray\n" );
 233          fwrite( $fp1, "Representation (all, hydrophobic) .................... crep= all\n" );
 234          fwrite( $fp1, "Number of matches to output .......................... maxm= 1000\n" );
 235          fwrite( $fp1, "Angle for rotations, deg (10,12,15,18,20,30, 0-no rot.)  ai= 20\n" );    
 236          }else
 237          {
 238              fwrite( $fp1, "Matching mode (generic/helix) ....................... mmode= generic\n" );
 239              fwrite( $fp1, "Grid step ............................................. eta= 1.7\n" );
 240              fwrite( $fp1, "Repulsion (attraction is always -1) .................... ro= 30.\n" );
 241              fwrite( $fp1, "Attraction double range (fraction of single range) ..... fr= 0.\n" );
 242              fwrite( $fp1, "Potential range type (atom_radius, grid_step) ....... crang= atom_radius\n" );
 243               fwrite( $fp1, "Projection (blackwhite, gray) ................ ....... ccti= gray\n" );
 244               fwrite( $fp1, "Representation (all, hydrophobic) .................... crep= all\n" );
 245              fwrite( $fp1, "Number of matches to output .......................... maxm= 1000\n" );
 246              fwrite( $fp1, "Angle for rotations, deg (10,12,15,18,20,30, 0-no rot.)  ai= 10\n" );
 247          }
 248  
 249  fclose( $fp1 );
 250  
 251  // Writing wlist.gr
 252  $wlist="$user_wd_path/tinker/wlist.gr";
 253  
 254  $fp2 = fopen( "$wlist", "w" );
 255  
 256      fwrite( $fp2, "# File_of_predictions   First_match   Last_match   separate/joint  +init_lig\n" );
 257      fwrite( $fp2, "# ----------------------------------------------------------------------------\n" );
 258      fwrite( $fp2, "$receptor-$ligand.res    1   10    separ    no\n" );    
 259  
 260  fclose( $fp2 );
 261  }

 293  // The results directory generated by the grid middleware starts with the grid user name
 294  $dir = dir(".");    
 295  $dir->rewind();
 296      while ($file = $dir->read())
 297      {    
 298          // To change: $user
 299          if ( substr($file, 0, 5)=="$server_user" )
 300          {
 301              // We extract the gramm results
 302              exec("tar -zxvf $file/gramm-come.tar.gz -C .");
